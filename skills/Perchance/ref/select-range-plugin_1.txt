$output (list, start, end) =>
	let result;
	let arr = list.selectAll;
	
	if(start < 0) { 
		start += 1;
		if(start === 0) return [arr[arr.length-1]]; // last item only
	}
	
	if(end < 0) {
		end += 1;
		if(end === 0) end = undefined; // all the way to the end
	}
	
	if(start === 0 || (start > 0 && end > 0 && start > end)) return [`(invalid inputs to select-range-plugin: ${start}${end === undefined ? '' : ","+end})`];
	if(end) {
		if(start > 0) result = arr.slice(start-1, end);
		else result = arr.slice(start, end);
	} else {
		result = arr.slice(start-1);
	}

  // Make it behave like a list when stringified
  result.toString = function() {
    if(this.length === 0) return "";
    return this.selectOne+"";
  };
  
	return result;
