<script>
  // ================= 核心修复：声明纯 JS 全局变量 =================
  var finalPrompt_JS = "(masterpiece:1.2), (best quality:1.2), (8k resolution:1.1).";

  // ================= 配置区 =================
  const DIFY_KEY = "app-0Nzs538btupJWHvFVCpHZFqP";
  const DIFY_URL = "https://aichat.zerone.us.kg/v1";
  // ==========================================

  async function runRealityFlow() {
    const fileNode = document.getElementById('img-file');
    const queryNode = document.getElementById('user-query');
    const statusNode = document.getElementById('process-status');
    const btn = document.getElementById('gen-btn');

    const hasImage = !!fileNode.files[0];
    const userText = queryNode.value.trim();

    // 最小化拦截校验：图和文不能同时为空
    if (!hasImage && !userText) {
      return alert("请至少上传参考图，或者输入提示词！");
    }

    try {
      btn.disabled = true;
      let uploadedFilesArray = []; // 用于动态传递给 Dify

      // 最小化分流：如果有图，才执行上传
      if (hasImage) {
        statusNode.innerText = "⏳ [图生图] 正在上传图片基因...";

        const formData = new FormData();
        formData.append('file', fileNode.files[0]);
        formData.append('user', 'reality_web_user');

        const upRes = await fetch(`${DIFY_URL}/files/upload`, {
          method: 'POST',
          headers: { 'Authorization': `Bearer ${DIFY_KEY}` },
          body: formData
        });
        
        const upData = await upRes.json();
        if (!upRes.ok) throw new Error(upData.message || "图片上传失败");

        uploadedFilesArray.push({
          "type": "image",
          "transfer_method": "local_file",
          "upload_file_id": upData.id
        });
        
        statusNode.innerText = "🧬 [图生图] 正在执行视觉解构与缝合...";
      } else {
        statusNode.innerText = "✍️ [文生图] 正在呼叫增强大师裂变提示词...";
      }

      const wfRes = await fetch(`${DIFY_URL}/chat-messages`, {
        method: 'POST',
        headers: { 
          'Authorization': `Bearer ${DIFY_KEY}`,
          'Content-Type': 'application/json' 
        },
        body: JSON.stringify({
          inputs: { 
            "query": userText || "highly detailed",
            "files": uploadedFilesArray // 动态数组，空则触发 Dify 文生图逻辑
          },
          query: userText || "Run visual analysis",
          user: "reality_web_user",
          response_mode: "blocking"
        })
      });
      
      const resultData = await wfRes.json();
      
      if (!wfRes.ok) {
        throw new Error(resultData.message || "工作流执行失败");
      }

      let rawAnswer = resultData.answer;
      if (!rawAnswer) {
        throw new Error("未能获取到缝合提示词");
      }

      // ==========================================
      // 🚀 原封不动保留的核心逻辑：数据清洗与自动画幅推断
      // ==========================================
      let cleanPrompt = rawAnswer.replace(/```[a-z]*\n?/gi, ''); 
      cleanPrompt = cleanPrompt.replace(/`/g, '');

      // 捕获大模型传回的智能比例暗号
      let arMatch = cleanPrompt.match(/\[AR:([pls])\]/i);
      if (arMatch) {
        let arType = arMatch[1].toLowerCase();
        // 动态覆盖左侧的 Perchance 变量
        if (arType === 'p') { window.imgWidth = 512; window.imgHeight = 768; }
        else if (arType === 'l') { window.imgWidth = 768; window.imgHeight = 512; }
        else { window.imgWidth = 512; window.imgHeight = 512; }
        
        console.log("📐 智能大脑已自动调整画布比例为:", arType);
        
        // 擦除暗号，避免污染提示词
        cleanPrompt = cleanPrompt.replace(/\[AR:[pls]\]/i, '');
      } else {
        // 兜底默认值：横图
        window.imgWidth = 768; window.imgHeight = 512;
      }

      cleanPrompt = cleanPrompt.replace(/^(Here is the prompt:|Generated Prompt:|最终缝合提示词：?)\s*/i, '');
      cleanPrompt = cleanPrompt.replace(/\n+/g, ' ');
      cleanPrompt = cleanPrompt.trim();

      statusNode.innerText = "✨ 洗净完成！召唤 FLUX 引擎中...";
      console.log("【洗净后的最终纯粹提示词】:", cleanPrompt);

      // --- 终极注入并更新视图 ---
      finalPrompt_JS = cleanPrompt;
      window.update();

    } catch (err) {
      console.error(err);
      statusNode.innerHTML = `<span style="color:#ff6b6b">❌ 链路中断: ${err.message}</span>`;
    } finally {
      btn.disabled = false;
    }
  }
</script>
