    <h2>Step 1: What is Perchance?</h2>
    <p>Perchance is a platform for creating random text generators. Users can create generators with various outputs and options, and these generators can be accessed via an API for integration with other applications.</p>
    <p>Visit <a href="https://perchance.org" target="_blank">perchance.org</a> to explore available generators or create your own.</p>
  </div>
  
  <div class="step">
    <h2>Step 2: Finding a Generator to Use</h2>
    <p>Browse Perchance for a generator you'd like to use, or create your own. For this tutorial, we'll use a simple example generator.</p>
    <p>When you find a generator, note its URL. The format is typically: <code>https://perchance.org/generator-name</code></p>
  </div>
  
  <div class="step">
    <h2>Step 3: Understanding the Perchance API</h2>
    <p>Perchance offers a simple API to access generator results. The basic format is:</p>
    <pre>https://perchance.org/api/generateList.php?generator=GENERATOR_NAME&count=NUMBER_OF_RESULTS</pre>
    <p>For example, to get 3 results from a generator called "example-generator":</p>
    <pre>https://perchance.org/api/generateList.php?generator=example-generator&count=3</pre>
    <p>This returns a JSON array of generated results.</p>
  </div>

    <p>You might encounter CORS (Cross-Origin Resource Sharing) issues when fetching directly from Perchance. If this happens, you have a few options:</p>
    <ol>
      <li>Use a CORS proxy service</li>
      <li>Create a simple backend server to make the request for you</li>
      <li>Contact Perchance to request CORS access for your domain</li>
    </ol>

[Route("api/[controller]")]
public class ProxyController : ControllerBase
{
    private readonly HttpClient _httpClient;
    
    public ProxyController(HttpClient httpClient)
    {
        _httpClient = httpClient;
    }
    
    [HttpGet]
    public async Task&lt;IActionResult&gt; ProxyPerchance(string generator, int count)
    {
        try
        {
            string url = $"https://perchance.org/api/generateList.php?generator={generator}&count={count}";
            HttpResponseMessage response = await _httpClient.GetAsync(url);
            
            if (!response.IsSuccessStatusCode)
                return StatusCode((int)response.StatusCode);
                
            var content = await response.Content.ReadAsStringAsync();
            return Content(content, "application/json");
        }
        catch
        {
            return StatusCode(500);
        }
    }
}</pre>

    def fetch_from_perchance(self):
        def fetch_task():
            generator_name = self.generator_input.get()
            count = 1
            
            try:
                self.root.after(0, lambda: self.generate_button.config(
                    text="Loading...", 
                    state="disabled"
                ))
                
                url = f"https://perchance.org/api/generateList.php?generator={generator_name}&count={count}"
                response = requests.get(url)
                
                if response.status_code != 200:
                    raise Exception(f"HTTP error! Status: {response.status_code}")
                
                data = response.json()
                
                if data and len(data) > 0:
                    self.root.after(0, lambda: self.output_text.config(text=data[0]))
                else:
                    self.root.after(0, lambda: self.output_text.config(text="No results returned."))
            
            except Exception as e:
                print(f"Error fetching from Perchance: {str(e)}")
                self.root.after(0, lambda: messagebox.showerror("Error", str(e)))
                self.root.after(0, lambda: self.output_text.config(
                    text="Error fetching from Perchance. Check console for details."
                ))
            
            finally:
                self.root.after(0, lambda: self.generate_button.config(
                    text="Generate", 
                    state="normal"
                ))
        
        # Start the fetch task in a separate thread to avoid freezing the UI
        threading.Thread(target=fetch_task).start()
