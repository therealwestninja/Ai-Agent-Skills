<p style="margin-top:0;">This plugin allows you to easily add an image to your generator. Simply import the plugin by putting this code on a new line in the list editor (the panel on the left):</p>
  <p>["&lt;p>"+body.joinItems("&lt;/p>&lt;p>")+"&lt;/p>"]</p>
<br>
	<p><b>Notes:</b></p>
	<ul>
		["&lt;li>"+notes.joinItems("&lt;/li>&lt;li>")+"&lt;/li>"]
	</ul>

<script>
  function code(text) {
    return `<pre>${text}</pre>`
  }
  
  function ulist(l) {
    return "<li>"+l.joinItems("</li><li>")+"</li>"
  }
  
  function olist(l) {
    return "<ol><li>"+l.joinItems("</li><li>")+"</li></ol>"
  }
  
  function link(url,text) {
    return `<a href="${url}" target="_blank">${text}</a>`
  }
  
  function bold(text) {
    return `<b>${text}</b>`
  }
  
  function italic(text) {
    return `<i>${text}</i>`
  }
  
  function underline(text) {
    return `<u>${text}</u>`
  }
  
  function color(text, color) {
    return `<span style="color:${color};">${text}</span>`
  }
  
  function inline_code(text) {
    return `<span style="display:inline;text-align:left;background: #333;color: white;padding: 0.3em;border-radius: 2px;tab-size: 2;		-moz-tab-size: 2;		-o-tab-size: 2;-webkit-tab-size: 2;">${text}</span>`
  }
</script>
