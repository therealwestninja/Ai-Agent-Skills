headerElementClickHandler(headerEl) =>
  let rootNode = window["nested_plugin_root_node_global_ref_822467"];
	let isLeaf = !rootNode[headerEl.dataset.archetypeId];
	if(isLeaf) return;
  let wrapperEl = headerEl.closest('.item-wrapper');
  let bodyEl = wrapperEl.querySelector(".item-body");
	let marginEl = wrapperEl.querySelector('.item-margin');
	if(bodyEl) {
	  let marginEl = wrapperEl.querySelector('.item-margin');
	  if(bodyEl.style.display === "") { bodyEl.style.display = "none"; marginEl.innerHTML = "➕&#xFE0E;"; }
		else if(bodyEl.style.display === "none") { bodyEl.style.display = ""; marginEl.innerHTML = "➖&#xFE0E;"; }
	} else {
		let tmp = document.createElement('div');
		tmp.innerHTML = generateItemBody(headerEl.dataset.archetypeId);
		let bodyEl = tmp.firstChild;
		wrapperEl.querySelector('.item-main').appendChild(bodyEl);
		marginEl.innerHTML = "➖&#xFE0E;";
	}


generateItemBody(archetypeId) => 
  let rootNode = window["nested_plugin_root_node_global_ref_822467"];
	let isLeaf = !rootNode[archetypeId] || rootNode[archetypeId].selectAll.length === 0;
	if(isLeaf && rootNode[archetypeId].description) return `<div class='item-body'>${rootNode[archetypeId].description}</div>`;
	if(isLeaf) return `<div class='item-body'></div>`;
	let node = rootNode[archetypeId];
  node = node.getSelf;
	let specifiedOverallCount = Number(node.n) || null;
	let bodyText = "";
	let currentOverallCount = 0;
	let children = node.selectAll;
	let finishedChildNames = new Set();
	for(let child of children) {
		let [name, quantity] = child.evaluateItem.trim().replace(/\s+/," ").split(" ").slice(0, 2);
		if(!quantity) continue;
		for(let i = 0; i < Number(quantity); i++) {
	    bodyText += generateItem(name);
			currentOverallCount++;
		}
		finishedChildNames.add(name);
	}
	if(specifiedOverallCount && finishedChildNames.size < children.length && currentOverallCount < specifiedOverallCount) {
		for(let i = 0; i < 10000; i++) {
			let name = node.selectOne.evaluateItem.trim().replace(/\s+/," ").split(" ")[0];
			if(finishedChildNames.has(name)) continue;
			else {
				bodyText += generateItem(name);
				currentOverallCount++;
				if(currentOverallCount >= specifiedOverallCount) break;
			}
		}
	}
	if(currentOverallCount === 0) { // they didn't specify an overall count or individual counts, so we grab 5-10 random items as a sane default
		let defaultQuantity = 5+Math.round(Math.random()*5);
		for(let i = 0; i < defaultQuantity; i++) {
		  let name = node.selectOne.evaluateItem.trim().replace(/\s+/," ").split(" ")[0];
	    bodyText += generateItem(name);
		}
	}
  return `<div class='item-body'>${bodyText}</div>`;
