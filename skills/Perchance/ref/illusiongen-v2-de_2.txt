        // Import Tippy.js for tooltips
        import tippy from 'https://esm.sh/tippy.js@6.3.7';

        // --- SeedBatchQueue Class (Converted from Python) ---
        class SeedBatchQueue {
            constructor(artStylesList) {
                this.batchList = [];
                this.current_index = 0;
                this.loop_mode = false;
                this.art_styles = [...artStylesList];
                this.selected_styles = [...artStylesList];
                this.style_loop = false;
                this.hide_styles = false;
                this.current_style_index = 0;
            }

            add_seed(seed_number) {
                this.batchList.push(seed_number);
            }

            remove_seed(index) {
                if (index >= 0 && index < this.batchList.length) {
                    this.batchList.splice(index, 1);
                    if (this.current_index >= index && this.current_index > 0) {
                        this.current_index--;
                    }
                }
            }

            duplicate_seed(index) {
                if (index >= 0 && index < this.batchList.length) {
                    this.batchList.splice(index + 1, 0, this.batchList[index]);
                }
            }

            get_next_seed() {
                if (this.batchList.length === 0) return null;

                const seed = this.batchList[this.current_index];
                this.current_index++;

                if (this.loop_mode) {
                    if (this.current_index >= this.batchList.length) {
                        this.current_index = 0;
                    }
                } else if (this.current_index >= this.batchList.length) {
                    return null;
                }

                return seed;
            }

            toggle_art_style(style_name) {
                const index = this.selected_styles.indexOf(style_name);
                if (index > -1) {
                    this.selected_styles.splice(index, 1);
                } else {
                    this.selected_styles.push(style_name);
                }
            }

            get_next_style() {
                if (this.selected_styles.length === 0) return null;

                if (!this.style_loop) {
                    return this.selected_styles[0];
                }

                const style = this.selected_styles[this.current_style_index % this.selected_styles.length];
                this.current_style_index++;
                return style;
            }

            toggle_hide_styles() {
                this.hide_styles = !this.hide_styles;
            }

            reset() {
                this.current_index = 0;
                this.current_style_index = 0;
            }

            clear() {
                this.batchList = [];
                this.current_index = 0;
                this.current_style_index = 0;
            }
        }

        function updateYinYangAnimation() {
            if (!document.getElementById('loadingOverlayEl').classList.contains('active')) {
                yinYangAnimationId = null;
                return;
            }
            
            const loader = document.getElementById('yinYangLoader');
            
            // Update speed (accelerate until target, then decelerate and reverse)
            if (yinYangAccelerating) {
                yinYangSpeed = Math.min(yinYangSpeed + 0.01, yinYangTargetSpeed / 1000);
                if (yinYangSpeed >= yinYangTargetSpeed / 1000) {
                    yinYangAccelerating = false;
                }
            } else {
                yinYangSpeed = Math.max(yinYangSpeed - 0.01, -yinYangTargetSpeed / 1000);
                if (yinYangSpeed <= -yinYangTargetSpeed / 1000) {
                    yinYangAccelerating = true;
                    yinYangDirection = -yinYangDirection; // Reverse direction
                }
            }
            
            // Apply animation
            const duration = Math.abs(1 / yinYangSpeed);
            loader.style.setProperty('--duration', `${duration}s`);
            loader.style.animationDirection = yinYangSpeed > 0 ? 'normal' : 'reverse';
            
            yinYangAnimationId = requestAnimationFrame(updateYinYangAnimation);
        }

        function startYinYangAnimation() {
            if (!yinYangAnimationId) {
                yinYangSpeed = 0.12;
                yinYangAccelerating = true;
                yinYangDirection = 1;
                updateYinYangAnimation();
            }
        }

        function stopYinYangAnimation() {
            if (yinYangAnimationId) {
                cancelAnimationFrame(yinYangAnimationId);
                yinYangAnimationId = null;
            }

        // --- Touch Gesture Handling ---
        const imageDisplayCtn = document.getElementById('image-display-ctn');
        const touchHintEl = document.getElementById('touchHintEl');
        let touchHintTimeout;

        function showTouchHint() {
            touchHintEl.classList.add('visible');
            clearTimeout(touchHintTimeout);
            touchHintTimeout = setTimeout(() => {
                touchHintEl.classList.remove('visible');
            }, 3000);
        }

        imageDisplayCtn.addEventListener('touchstart', function(e) {
            if (e.touches.length === 1) {
                touchStartX = e.touches[0].clientX;
                touchStartY = e.touches[0].clientY;
            } else if (e.touches.length === 2) {
                // Pinch start
                initialPinchDistance = Math.hypot(
                    e.touches[0].clientX - e.touches[1].clientX,
                    e.touches[0].clientY - e.touches[1].clientY
                );
                e.preventDefault();
            }
        }, { passive: false });

        imageDisplayCtn.addEventListener('touchmove', function(e) {
            if (e.touches.length === 2) {
                // Pinch zoom
                const currentPinchDistance = Math.hypot(
                    e.touches[0].clientX - e.touches[1].clientX,
                    e.touches[0].clientY - e.touches[1].clientY
                );
                const scaleChange = currentPinchDistance / initialPinchDistance;
                currentScale = Math.min(Math.max(currentScale * scaleChange, 0.5), 3);
                previewImgEl.style.transform = `scale(${currentScale})`;
                previewScaleSlider.value = currentScale * 100;
                initialPinchDistance = currentPinchDistance;
                e.preventDefault();
            }
        }, { passive: false });

        imageDisplayCtn.addEventListener('touchend', function(e) {
            if (e.changedTouches.length === 1 && Math.abs(currentScale - 1) < 0.1) {
                const touchEndX = e.changedTouches[0].clientX;
                const touchEndY = e.changedTouches[0].clientY;
                
                const diffX = touchEndX - touchStartX;
                const diffY = touchEndY - touchStartY;
                
                // Detect swipe (horizontal movement greater than vertical)
                if (Math.abs(diffX) > 50 && Math.abs(diffX) > Math.abs(diffY)) {
                    if (diffX > 0) {
                        // Swipe right - previous image
                        navigateGallery(-1);
                    } else {
                        // Swipe left - next image
                        navigateGallery(1);
                    }
                    showTouchHint();
                }
            }
        });

            loadToPreview(galleryImages[currentGalleryIndex].url);
        }

        // Generate Image Logic (Batch Support - Properly Sequential with Unique Seeds)
        window.triggerGenerate = async function() {
            if (isGenerating) return;
            if (document.getElementById('generateBtn').disabled) return;
            
            isGenerating = true;
            stopRequested = false;
            
            const batchSizeVal = parseInt(document.getElementById('batchSizeDisplay').textContent);
            
            // Reset queue index if starting fresh
            if (seedBatchQueue.current_index === 0 || seedBatchQueue.current_index >= seedBatchQueue.batchList.length) {
                seedBatchQueue.reset();
            }
            
            document.getElementById('generateBtn').disabled = true;
            document.getElementById('fabGenerateBtn').disabled = true;
            document.getElementById('stopBtn').classList.add('active');
            
            const total = batchSizeVal;
            const progressBar = document.getElementById('batchProgressBar');
            const progressFill = document.getElementById('batchProgressFill');
            
            if (total > 1) {
                progressBar.classList.add('active');
            }
            
            // Start Yin-Yang animation
            startYinYangAnimation();
            
            let currentIndex = 1;
            
            while (currentIndex <= total && !stopRequested) {
                progressFill.style.width = `${((currentIndex - 1) / total) * 100}%`;
                
                await new Promise(r => setTimeout(r, 50)); // UI Breath
                
                await generateSingleImage(currentIndex, total);
                
                if (stopRequested) break;
                
                progressFill.style.width = `${(currentIndex / total) * 100}%`;
                
                if (currentIndex < total) {
                    document.getElementById('statusText').textContent = currentLang === 'en' ? "Pausing before next image..." : "Pause vor dem nächsten Bild...";
                    await new Promise(r => setTimeout(r, 800));
                }
                currentIndex++;
            }
            
            isGenerating = false;
            document.getElementById('generateBtn').disabled = false;
            document.getElementById('fabGenerateBtn').disabled = false;
            document.getElementById('stopBtn').classList.remove('active');
            progressBar.classList.remove('active');
            progressFill.style.width = '0%';
            document.getElementById('loadingOverlayEl').classList.remove('active');
            stopYinYangAnimation();
            
            if (!stopRequested) {
                document.getElementById('statusText').textContent = currentLang === 'en' 
                    ? `Batch complete! Generated ${currentIndex - 1} image${(currentIndex - 1) > 1 ? 's' : ''}.`
                    : `Stapel komplett! ${currentIndex - 1} Bild${(currentIndex - 1) > 1 ? 'er' : ''} generiert.`;
            }
        }

        async function generateSingleImage(currentIndex, total) {
            const isBatch = total > 1;
            const statusText = isBatch 
                ? (currentLang === 'en' ? `Generating ${currentIndex}/${total}...` : `Generiere ${currentIndex}/${total}...`) 
                : i18n[currentLang].generatingIllusion;
            
            document.getElementById('statusText').textContent = statusText;
            document.getElementById('loadingOverlayEl').classList.add('active');
            document.getElementById('loadingTextEl').textContent = isBatch 
                ? (currentLang === 'en' ? `Image ${currentIndex} of ${total}` : `Bild ${currentIndex} von ${total}`) 
                : i18n[currentLang].generatingIllusion;
            document.getElementById('batchStatusText').textContent = isBatch ? (currentLang === 'en' ? "Processing sequentially..." : "Verarbeite sequenziell...") : "";

            try {
                let prompt = document.getElementById('promptInput').value;
                const negative = document.getElementById('negativePromptInput').value;
                let style = document.getElementById('styleSelect').value;
                const flags = document.getElementById('flagsInput').value;

                // Check for randomize style
                const randomizeStyle = document.getElementById('randomizeStyleCheck').checked;
                if (randomizeStyle) {
                    style = artStyles[Math.floor(Math.random() * artStyles.length)];
                }

                // Check for seed queue
                let uniqueSeed = seedBatchQueue.get_next_seed();
                if (uniqueSeed === null) {
                    // No more seeds in queue, use default logic
                    uniqueSeed = appSettings.seed > -1 ? appSettings.seed + currentIndex - 1 : Math.floor(Math.random() * 999999999);
                }

                // Check for style cycling with seed queue
                if (seedBatchQueue.style_loop && seedBatchQueue.batchList.length > 0) {
                    const nextStyle = seedBatchQueue.get_next_style();
                    if (nextStyle) {
                        style = nextStyle;
                    }
                }

                // Check for randomize effects
                const randomizeEffects = document.getElementById('randomizeEffectsCheck').checked;
                let currentEffects = [...selectedEffects];
                if (randomizeEffects) {
                    const numEffects = Math.floor(Math.random() * 8) + 1; // 1-8 effects
                    currentEffects = [];
                    const shuffled = [...effectsList].sort(() => 0.5 - Math.random());
                    for (let i = 0; i < numEffects && i < shuffled.length; i++) {
                        currentEffects.push(shuffled[i]);
                    }
                }

                // Construct full prompt
                let fullPrompt = `Optical illusion, ${style}, ${prompt}`;
                if (currentEffects.length > 0) {
                    fullPrompt += `, effects: ${currentEffects.join(', ')}`;
                }
                // Add unique seed marker to prompt
                fullPrompt += ` --seed ${uniqueSeed}`;

                let resolution = selectedResolution;
                const flagsLower = flags.toLowerCase();
                
                if (flagsLower.includes("--ar 3:2") || flagsLower.includes("--ar 1.5")) resolution = "768x512";
                else if (flagsLower.includes("--ar 2:3")) resolution = "512x768";
                else if (flagsLower.includes("--ar 16:9") || flagsLower.includes("--ar 1.77")) resolution = "768x432";
                else if (flagsLower.includes("--ar 9:16") || flagsLower.includes("--ar 0.56")) resolution = "432x768";

                // Build generation options object
                let genOptions = {
                    prompt: fullPrompt,
                    resolution: resolution,
                    count: 1
                };
                
                if(appSettings.model && appSettings.model !== 'default') genOptions.model = appSettings.model;
                if(appSettings.sampler) genOptions.sampler = appSettings.sampler;
                if(appSettings.steps) genOptions.steps = appSettings.steps;
                if(appSettings.cfg) genOptions.cfg_scale = appSettings.cfg;

                // GENERATE IMAGE using object-based API
                let dataUrl = await root.generateImage(genOptions);
                
                if (!dataUrl) throw new Error("No image data returned");
                
                currentImageUrl = dataUrl;
                document.getElementById('previewImgEl').src = dataUrl;
                currentScale = 1; // Reset zoom
                previewScaleSlider.value = 100;
                
                // Store metadata for export
                const imageMetadata = {
                    id: Date.now() + currentIndex,
                    url: dataUrl,
                    prompt: prompt,
                    negative: negative,
                    style: style,
                    effects: [...currentEffects],
                    flags: flags,
                    resolution: resolution,
                    seed: uniqueSeed,
                    model: appSettings.model,
                    sampler: appSettings.sampler,
                    steps: appSettings.steps,
                    cfg: appSettings.cfg,
                    batchIndex: `${currentIndex}/${total}`,
                    timestamp: new Date().toISOString()
                };
                
                await saveToHistory(prompt, negative, style, currentEffects, flags, resolution);
                
                if (isBatch) {
                    try {
                        let gallery = await root.kv.gallery.get("images") || [];
                        gallery.unshift(imageMetadata);
                        await root.kv.gallery.set("images", gallery);
                        galleryImages = gallery; // Update for touch navigation
                    } catch (err) {
                        console.error("Gallery Save Error:", err);
                    }
                }
                
                renderSeedQueue(); // Update queue UI
                
                document.getElementById('statusText').textContent = isBatch 
                    ? (currentLang === 'en' ? `Completed ${currentIndex}/${total}` : `Fertig ${currentIndex}/${total}`) 
                    : (currentLang === 'en' ? 'Generated successfully.' : 'Erfolgreich generiert.');

            } catch (err) {
                console.error(err);
                document.getElementById('statusText').textContent = currentLang === 'en' ? 'Error generating image.' : 'Fehler bei der Generierung.';
                document.getElementById('loadingOverlayEl').classList.remove('active');
            }
        }

        document.getElementById('exportZipBtn').addEventListener('click', async function() {
            const gallery = await root.kv.gallery.get("images") || [];
            if (gallery.length === 0) {
                document.getElementById('statusText').textContent = currentLang === 'en' ? 'No images to export.' : 'Keine Bilder zum Exportieren.';
                return;
            }

            this.disabled = true;
            this.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Creating ZIP...';
            document.getElementById('statusText').textContent = currentLang === 'en' ? 'Creating ZIP archive...' : 'Erstelle ZIP-Archiv...';

            try {
                const zip = new JSZip();
                const metadataFolder = zip.folder("metadata");

                for (let i = 0; i < gallery.length; i++) {
                    const item = gallery[i];
                    const imageIndex = i + 1;
                    
                    // Convert data URL to blob
                    const response = await fetch(item.url);
                    const blob = await response.blob();
                    
                    // Add image to ZIP
                    const filename = `image_${String(imageIndex).padStart(4, '0')}_${item.id}.png`;
                    zip.file(filename, blob);
                    
                    // Add metadata file
                    const metadata = {
                        filename: filename,
                        id: item.id,
                        prompt: item.prompt,
                        negative: item.negative || '',
                        style: item.style,
                        effects: item.effects || [],
                        flags: item.flags || '',
                        resolution: item.resolution,
                        seed: item.seed,
                        model: item.model,
                        sampler: item.sampler,
                        steps: item.steps,
                        cfg: item.cfg,
                        batchIndex: item.batchIndex,
                        timestamp: item.timestamp,
                        recreationPrompt: `Optical illusion, ${item.style}, ${item.prompt}${item.effects && item.effects.length > 0 ? ', effects: ' + item.effects.join(', ') : ''} --seed ${item.seed} --ar ${item.resolution.split('x')[0]/item.resolution.split('x')[1]}`
                    };
                    metadataFolder.file(`${filename.replace('.png', '.json')}`, JSON.stringify(metadata, null, 2));
                }

                // Add README
                const readme = `IllusionGen Gallery Export
Generated: ${new Date().toISOString()}
Total Images: ${gallery.length}

This ZIP contains all images from your gallery along with their complete metadata.

Each image has a corresponding JSON file in the 'metadata/' folder containing:
- All generation parameters (prompt, negative, style, effects, etc.)
- Model settings (model, sampler, steps, CFG scale)
- Seed value for reproduction
- A pre-formatted 'recreationPrompt' that can be used to regenerate the exact same image

To recreate an image:
1. Extract the ZIP
2. Open the corresponding JSON file for the desired image
3. Copy the 'recreationPrompt' value
4. Paste it into IllusionGen's prompt field
5. Set the seed to the value in the JSON file
6. Generate the image
`;
                zip.file("README.txt", readme);

                // Generate ZIP
                const content = await zip.generateAsync({ type: "blob" });
                const downloadUrl = URL.createObjectURL(content);
                
                const downloadAnchor = document.createElement('a');
                downloadAnchor.href = downloadUrl;
                downloadAnchor.download = `illusion-gen-gallery-${Date.now()}.zip`;
                document.body.appendChild(downloadAnchor);
                downloadAnchor.click();
                downloadAnchor.remove();
                URL.revokeObjectURL(downloadUrl);

                document.getElementById('statusText').textContent = currentLang === 'en' ? 'ZIP exported successfully!' : 'ZIP erfolgreich exportiert!';
            } catch (err) {
                console.error("ZIP Export Error:", err);
                document.getElementById('statusText').textContent = currentLang === 'en' ? 'Error creating ZIP.' : 'Fehler beim Erstellen des ZIP.';
            } finally {
                this.disabled = false;
                this.innerHTML = '<i class="fa-solid fa-file-zipper"></i> ' + (currentLang === 'en' ? 'Export ZIP' : 'ZIP exportieren');
            }
        });

        function renderGallery(items) {
            const grid = document.getElementById('galleryGridEl');
            grid.innerHTML = '';
            if (items.length === 0) {
                grid.innerHTML = `<div style="grid-column:1/-1; text-align:center; padding:50px;">${currentLang === 'en' ? 'No images saved yet.' : 'Noch keine Bilder gespeichert.'}</div>`;
                return;
            }
            items.forEach((item, index) => {
                let div = document.createElement('div');
                div.className = 'gallery-item';
                div.dataset.index = index; // Store index for navigation
                
                let batchIndicator = item.batchIndex ? `<span style="position:absolute; bottom:8px; left:8px; background:rgba(0,0,0,0.8); color:#fff; padding:3px 8px; border-radius:4px; font-size:0.75em; border:1px solid #444;">${item.batchIndex}</span>` : '';
                
                div.innerHTML = `
                    <img src="${item.url}" loading="lazy">
                    ${batchIndicator}
                    <button class="delete-btn" data-id="${item.id}"><i class="fa-solid fa-trash"></i></button>
                `;
                div.onclick = (e) => {
                    if(!e.target.closest('.delete-btn')) {
                        currentGalleryIndex = index;
                        loadToPreview(item.url);
                    }
                };
                grid.appendChild(div);
            });
            
            // Update grid based on current columns setting
            const columns = parseInt(document.getElementById('gridColumnsInput').value);
            updateGalleryGrid(columns);
            
            document.querySelectorAll('.delete-btn').forEach(btn => {
                btn.addEventListener('click', async (e) => {
                    e.stopPropagation();
                    if(!confirm(currentLang === 'en' ? 'Delete this image?' : 'Dieses Bild löschen?')) return;
                    let id = parseInt(btn.getAttribute('data-id'));
                    let gallery = await root.kv.gallery.get("images") || [];
                    gallery = gallery.filter(i => i.id !== id);
                    await root.kv.gallery.set("images", gallery);
                    galleryImages = gallery;
                    renderGallery(gallery);
                });
            });
        }

        // History Logic
        async function saveToHistory(prompt, negative, style, effects, flags, resolution) {
            try {
                let history = await root.kv.history.get("prompts") || [];
                const newEntry = {
                    id: Date.now(),
                    prompt: prompt,
                    negative: negative,
                    style: style,
                    effects: [...effects],
                    flags: flags,
                    resolution: resolution,
                    timestamp: new Date().toLocaleString()
                };
                
                if (history.length > 0 && history[0].prompt === prompt && history[0].style === style && history[0].resolution === resolution) return;
                
                history.unshift(newEntry);
                if (history.length > 100) history = history.slice(0, 100);
                await root.kv.history.set("prompts", history);
            } catch (err) {
                console.error("History Save Error:", err);
            }
        }

        function renderHistory(items) {
            const list = document.getElementById('historyListCtn');
            list.innerHTML = '';
            if (items.length === 0) {
                list.innerHTML = `<div style="text-align:center; padding:50px;">${currentLang === 'en' ? 'No prompt history yet.' : 'Noch keine Prompt-Historie.'}</div>`;
                return;
            }
            items.forEach(item => {
                let div = document.createElement('div');
                div.className = 'history-item';
                
                let effectsDisplay = item.effects && item.effects.length > 0 
                    ? `<div style="font-size:0.8em; color:#888; margin-top:6px;"><i class="fa-solid fa-wand-magic"></i> Effects: ${item.effects.join(', ')}</div>` 
                    : '';
                
                let resolutionDisplay = item.resolution 
                    ? `<span style="color:var(--accent); font-size:0.8em;">[${item.resolution}]</span>` 
                    : '';
                
                div.innerHTML = `
                    <div class="prompt-text">${resolutionDisplay} ${item.prompt}</div>
                    <div class="timestamp"><i class="fa-solid fa-clock"></i> ${item.timestamp} | <i class="fa-solid fa-palette"></i> ${item.style}</div>
                    ${effectsDisplay}
                    <div class="history-actions">
                        <button onclick="window.loadFromHistory(${item.id})" style="background:var(--accent); color:var(--bg-dark);"><i class="fa-solid fa-upload"></i> ${currentLang === 'en' ? 'Load' : 'Laden'}</button>
                        <button onclick="window.deleteFromHistory(${item.id})" style="background:#e53935;"><i class="fa-solid fa-trash"></i> ${currentLang === 'en' ? 'Delete' : 'Löschen'}</button>
                    </div>
                `;
                list.appendChild(div);
            });
        }
        
        window.loadFromHistory = async function(id) {
            let history = await root.kv.history.get("prompts") || [];
            let item = history.find(h => h.id === id);
            if (item) {
                document.getElementById('promptInput').value = item.prompt;
                document.getElementById('negativePromptInput').value = item.negative;
                document.getElementById('styleSelect').value = item.style;
                selectedEffects = [...item.effects];
                document.querySelectorAll('.effect-item').forEach(ei => {
                    ei.classList.toggle('selected', selectedEffects.includes(ei.getAttribute('data-effect')));
                });
                document.getElementById('flagsInput').value = item.flags;
                
                if (item.resolution) {
                    document.querySelectorAll('.format-btn').forEach(btn => {
                        const btnRes = btn.getAttribute('data-resolution');
                        if (btnRes === item.resolution) {
                            btn.click();
                        }
                    });
                }
                
                document.getElementById('historyModalEl').classList.remove('open');
                document.getElementById('statusText').textContent = currentLang === 'en' ? 'Loaded prompt from history.' : 'Prompt aus Verlauf geladen.';
            }
        };
        
        window.deleteFromHistory = async function(id) {
            if(!confirm(currentLang === 'en' ? 'Delete this history item?' : 'Diesen Eintrag löschen?')) return;
            let history = await root.kv.history.get("prompts") || [];
            history = history.filter(item => item.id !== id);
            await root.kv.history.set("prompts", history);
            renderHistory(history);
        };
