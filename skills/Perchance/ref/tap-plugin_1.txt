$output (listOrStr, style) =>
  let listReferenceId = "k" + Math.random().toString().slice(2) + Math.random().toString().slice(2);
  window[listReferenceId] = listOrStr;
  
  let initiallySelectedItem = listOrStr.evaluateItem;
  
  let html;
  if(style == undefined) html = `<span class="tap-id-${listReferenceId}" onclick="let result = window.${listReferenceId}.evaluateItem; document.querySelectorAll('.tap-id-${listReferenceId}').forEach(el => el.innerHTML=result);" style="cursor:pointer; font-weight:bold;">${initiallySelectedItem}</span>`;
  else if(style === "button")  html = `<button class="tap-id-${listReferenceId}" onclick="let result = window.${listReferenceId}.evaluateItem; document.querySelectorAll('.tap-id-${listReferenceId}').forEach(el => el.innerHTML=result);" style="cursor:pointer;">${initiallySelectedItem}</button>`;
  else html = `<span class="tap-id-${listReferenceId}" onclick="let result = window.${listReferenceId}.evaluateItem; document.querySelectorAll('.tap-id-${listReferenceId}').forEach(el => el.innerHTML=result);" style="${style || ""}">${initiallySelectedItem}</span>`
  
  let obj = {
    noTap: `<span class="tap-id-${listReferenceId}">${initiallySelectedItem}</span>`,
    noTapNoUpdate: initiallySelectedItem,
    html,
    toString: function() {
      return this.html;
    }
  };
  
  return obj;
