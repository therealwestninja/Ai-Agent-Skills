upload = {import:upload-plugin}

async handleClick() =>  // we must add 'async' before the function name because we're using 'await' in this function
  buttonEl.disabled = true;
  buttonEl.textContent = "uploading..."; 
  
  let { url, size } = await upload(inputTextEl.value);
  urlOutputEl.innerHTML = `<a href="${url}" target="_blank">${url}</a> (${(size/1024).toFixed(2)} kilobytes)`;
  
  buttonEl.textContent = "upload text";
  buttonEl.disabled = false;
