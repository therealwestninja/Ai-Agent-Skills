async $output(...args) =>
  let url;
  if(typeof args[0] === "object") url = args[0].href || args[0].url; // can be URL object, or {url, method, etc}
  else url = args[0];
  
  if(typeof url !== "string") {
    throw new Error("Invalid URL provided.");
  }

  if(url.startsWith("blob:") || url.startsWith("data:")) return window.fetch(...args);
  
  // Disallow relative URLs (for now, at least):
  if(!/^https?:\/\//.test(url)) {
    throw new Error("Must provide full URL, starting with https:// or http://");
  }

  let origin = new URL(url).origin;

  // for performance, exclude some CDNs that don't need CORS proxying
  if(origin.endsWith("jsdelivr.net") || origin.endsWith("catbox.moe") || (origin.endsWith("huggingface.co") && url.includes("/resolve/")) || origin === "https://raw.githubusercontent.com") {
    return window.fetch(...args);
  }
  // for some reason error/non-CORS responses get cached in the browser sometimes, so if that happens, we retry with proxy
  if(origin === "https://user-uploads.perchance.org" || origin === "https://user.uploads.dev" || origin === "https://aigc.uploads.dev") {
    let response = await window.fetch(...args).catch(e => {
      console.error(e);
      return false;
    });
    if(response !== false) return response;
  }
  
  // Note that we use `new URL(url).href` to exactly emulate the browser's encoding of 'invalid' URL characters (this is different to encodeURI IIUC). There were issues related to server handling of this sort of thing before I added this.
  // EDIT: Changed to putting URL in path instead of in query param for annoying/technical Cloudflare-related reasons.
  const proxiedUrl = `https://fetch-plugin.perchance.org/proxy1/${encodeURIComponent(new URL(url).href)}?origin=${encodeURIComponent(`https://${window.generatorPublicId}.perchance.org`)}`;
  if(typeof args[0] === "object" && args[0].url) { // <-- since it's possible to call fetch like this: fetch({url:"https://example.com"}) - i.e. options object as first param that *includes* the URL
    args[0] = new Request(proxiedUrl, args[0]);
    let response = await window.fetch(...args);
    if(response.status === 530) throw new Error("Failed to fetch"); // CF error for e.g. DNS lookup fail
    return response;
  } else {
    let response = await window.fetch(proxiedUrl, ...args.slice(1));
    if(response.status === 530) throw new Error("Failed to fetch"); // CF error for e.g. DNS lookup fail
    return response;
  }
