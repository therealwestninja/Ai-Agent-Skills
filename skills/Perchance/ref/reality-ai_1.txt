t2i = {import:text-to-image-plugin}

// 隐藏负面词库

// 生图基础参数
imgWidth = 512
imgHeight = 512
userScale = 7
userSeed = -1
batchCount = 4

// 核心渲染逻辑 (直接读取纯 JS 变量 finalPrompt_JS)
singleImage
  [t2i({prompt: finalPrompt_JS, negativePrompt: secretNeg, resolution: imgWidth + "x" + imgHeight, guidance: Number(userScale), seed: Number(userSeed)})]

batchOutput
  [singleImage.selectMany(Number(batchCount)).joinItems("")]
