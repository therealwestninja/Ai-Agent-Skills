  <!-- getGeneratorList Section -->

    </div>
    <div style="margin-bottom:10px;">
      <label style="display:block; margin-bottom:5px;">Tags (comma separated):</label>

    </div>
    <button id="listBtn" onclick="getGeneratorList()" style="padding:8px 15px; background:#2196F3; color:white; border:none; border-radius:4px; cursor:pointer;">Get List</button>
    <div id="listOutput" style="margin-top:15px; max-height:300px; overflow:auto;"></div>
  </div>

  <div style="margin-bottom:30px; padding:15px; border:1px solid #ddd; border-radius:8px;">

    <div style="margin-bottom:10px;">
      <label style="display:block; margin-bottom:5px;">Single generator name:</label>
      <input id="statsNameInput" type="text" placeholder="animal" style="width:100%; padding:8px;">
    </div>
    <div style="margin-bottom:10px;">
      <label style="display:block; margin-bottom:5px;">Multiple names (comma separated):</label>
      <input id="statsNamesInput" type="text" placeholder="animal,noun,adjective" style="width:100%; padding:8px;">

  <script>
    // Helper function to show loading state
    function setLoading(buttonEl, isLoading) {
      buttonEl.disabled = isLoading;
      buttonEl.textContent = isLoading ? "Loading..." : buttonEl.dataset.originalText;
    }

    // Initialize button original text
    document.querySelectorAll('button').forEach(btn => {
      btn.dataset.originalText = btn.textContent;
    });

    // Create generator card HTML
    function createGeneratorCard(generator) {
      return `<div class="generator-card">
        <h3>${generator.metaData.title || generator.name}</h3>
        <p><strong>Generator Name:</strong> ${generator.name}</p>
        <p><strong>Views:</strong> ${generator.views.toLocaleString()}</p>
        <p><strong>Last Edit Time:</strong> ${new Date(generator.lastEditTime).toLocaleString()}</p>
        <p><strong>Public ID:</strong> ${generator.publicId}</p>
        ${generator.metaData.description ? `<p><strong>Description:</strong> ${generator.metaData.description}</p>` : ''}
        ${generator.metaData.image ? `<img src="${generator.metaData.image}" style="max-width:100%; border-radius:4px; margin-top:10px;" />` : ''}
      </div>`;
    }

    // getGeneratorScreenshot
    async function getGeneratorScreenshot() {
      const btn = screenshotBtn;
      const output = screenshotOutput;
      const name = screenshotNameInput.value.trim();
      
      if (!name) {
        output.innerHTML = '<p style="color:red;">Please enter a generator name</p>';
        return;
      }
      
      setLoading(btn, true);
      output.innerHTML = '<p>Loading screenshot...</p>';
      
      try {
        const url = `https://perchance.org/api/getGeneratorScreenshot?generatorName=${encodeURIComponent(name)}`;
        const response = await superFetch(url);
        const blob = await response.blob();
        const imgUrl = URL.createObjectURL(blob);
        
        output.innerHTML = `<img src="${imgUrl}" style="max-width:100%; border-radius:4px;">`;
      } catch (error) {
        output.innerHTML = `<p style="color:red;">Error: ${error.message}</p>`;
      } finally {
        setLoading(btn, false);
      }
    }

    // getGeneratorList
    async function getGeneratorList() {
      const btn = listBtn;
      const output = listOutput;
      const max = listMaxInput.value.trim();
      const tags = listTagsInput.value.trim();
      
      setLoading(btn, true);
      output.innerHTML = '<p>Loading generator list...</p>';
      
      try {
        let url = 'https://perchance.org/api/getGeneratorList';
        const params = new URLSearchParams();
        
        if (max) params.append('max', max);
        if (tags) params.append('tags', tags);
        
        if (params.toString()) url += '?' + params.toString();
        
        const response = await superFetch(url);
        const data = await response.json();
        
        // Display generators as graphical cards
        if (data.generators && data.generators.length > 0) {
          let htmlContent = '';
          data.generators.forEach(generator => {
            htmlContent += createGeneratorCard(generator);
          });
          output.innerHTML = htmlContent;
        } else {
          output.innerHTML = '<p>No generators found</p>';
        }
      } catch (error) {
        output.innerHTML = `<p style="color:red;">Error: ${error.message}</p>`;
      } finally {
        setLoading(btn, false);
      }
    }

    // getGeneratorStats
    async function getGeneratorStats() {
      const btn = statsBtn;
      const output = statsOutput;
      const name = statsNameInput.value.trim();
      const names = statsNamesInput.value.trim();
      
      if (!name && !names) {
        output.innerHTML = '<p style="color:red;">Please enter at least one generator name</p>';
        return;
      }
      
      setLoading(btn, true);
      output.innerHTML = '<p>Loading generator stats...</p>';
      
      try {
        let url = 'https://perchance.org/api/getGeneratorStats?';
        
        if (names) {
          url += `names=${encodeURIComponent(names)}`;
        } else {
          url += `name=${encodeURIComponent(name)}`;
        }
        
        const response = await superFetch(url);
        const data = await response.json();
        
        if (data.status === 'success') {
          let htmlContent = '';
          if (Array.isArray(data.data)) {
            // Display multiple generators as a list
            data.data.forEach(generator => {
              htmlContent += createGeneratorCard(generator);
            });
          } else {
            // Display single generator without list container
            htmlContent = createGeneratorCard(data.data);
          }
          output.innerHTML = htmlContent;
        } else {
          output.innerHTML = `<p style="color:red;">Error: ${JSON.stringify(data)}</p>`;
        }
      } catch (error) {
        output.innerHTML = `<p style="color:red;">Error: ${error.message}</p>`;
      } finally {
        setLoading(btn, false);
      }
    }

    // downloadGenerator
    async function downloadGenerator() {
      const btn = downloadBtn;
      const output = downloadOutput;
      const name = downloadNameInput.value.trim();
      const listsOnly = listsOnlyCheck.checked;
      
      if (!name) {
        output.textContent = 'Please enter a generator name';
        output.style.display = 'block';
        return;
      }
      
      setLoading(btn, true);
      output.style.display = 'block';
      output.textContent = 'Preparing download...';
      
      try {
        let url = `https://perchance.org/api/downloadGenerator?generatorName=${encodeURIComponent(name)}`;
        if (listsOnly) url += '&listsOnly=true';
        
        const response = await superFetch(url);
        const blob = await response.blob();
        const downloadUrl = URL.createObjectURL(blob);
        
        const a = document.createElement('a');
        a.href = downloadUrl;
        a.download = `${name}${listsOnly ? '-lists' : ''}.html`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(downloadUrl);
        
        output.textContent = 'Download started!';
        setTimeout(() => { output.style.display = 'none'; }, 3000);
      } catch (error) {
        output.textContent = `Error: ${error.message}`;
      } finally {
        setLoading(btn, false);
      }
    }

    // getGeneratorsAndDependencies
    async function getGeneratorsAndDependencies() {
      const btn = depsBtn;
      const output = depsOutput;
      const names = depsNamesInput.value.trim();
      
      if (!names) {
        output.innerHTML = '<p style="color:red;">Please enter generator names</p>';
        return;
      }
      
      setLoading(btn, true);
      output.innerHTML = '<p>Loading generators and dependencies...</p>';
      
      try {
        const url = `https://perchance.org/api/getGeneratorsAndDependencies?generatorNames=${encodeURIComponent(names)}`;
        const response = await superFetch(url);
        const data = await response.json();
        
        // Always show graphical view
        buildDependencyGraph(data);
      } catch (error) {
        output.innerHTML = `<p style="color:red;">Error: ${error.message}</p>`;
      } finally {
        setLoading(btn, false);
      }
    }

    // Build graphical dependency view
    function buildDependencyGraph(data) {
      let html = '';
      
      // Process found generators
      if (data.generators) {
        for (let genName in data.generators) {
          let gen = data.generators[genName];
          html += `<div class="generator-deps-card">`;
          html += `<h4>${genName}</h4>`;
          
          if (gen.imports && gen.imports.length > 0) {
            html += '<ul>';
            gen.imports.forEach(imp => {
              html += `<li>${imp}</li>`;
            });
            html += '</ul>';
          } else {
            html += '<p>No imports</p>';
          }
          
          html += `</div>`;
        }
      }
      
      // Process unfound generators
      if (data.unfound && data.unfound.length > 0) {
        html += `<div class="unfound-generators">`;
        html += `<h4>Unfound Generators</h4>`;
        html += '<ul>';
        data.unfound.forEach(name => {
          html += `<li>${name}</li>`;
        });
        html += '</ul>';
        html += `</div>`;
      }
      
      depsOutput.innerHTML = html;
    }
  </script>
