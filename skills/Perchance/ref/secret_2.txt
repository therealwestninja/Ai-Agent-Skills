  (() => {
    const STORAGE_KEY = `secret-inbox:${window.generatorName}:v4`;
    const BLOG_SEED_KEY = `secret-inbox:${window.generatorName}:blog-seed:v4`;
    const PROFILE_THEME_IDS = ["harbor", "graphite", "cobalt", "olive", "ember", "rosewood", "coral", "sand", "dusk", "plum", "forest", "mint", "nightfall", "terracotta", "brass", "glacier", "cherry", "lavender", "noir", "citrus"];
    const URL_ID_ALPHABET = "123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz";
    const URL_ID_DIGITS = Object.create(null);
    for (let index = 0; index < URL_ID_ALPHABET.length; index += 1) {
      URL_ID_DIGITS[URL_ID_ALPHABET[index]] = index;
    }

    function normalizeId(value) {
      return String(value || "").trim().toLowerCase().replace(/[^0-9a-f]/g, "");
    }

    function isInternalId(value) {
      return /^[0-9a-f]{32}$/.test(String(value || ""));
    }

    function buildInboxChannel(inboxId) {
      const normalized = normalizeId(inboxId);
      return isInternalId(normalized) ? `${normalized}+ids:channel` : "";
    }

    function buildBlogChannel(inboxId, commentsPluginUsername) {
      const normalizedInboxId = normalizeId(inboxId);
      const username = sanitizeCommentsPluginUsername(commentsPluginUsername);
      return isInternalId(normalizedInboxId) && username ? `${normalizedInboxId}+u:${username},ids:channel` : "";
    }

    function encodeUrlId(value) {
      const hex = normalizeId(value);
      if (!isInternalId(hex)) return "";

      let leadingZeroBytes = 0;
      while (hex.slice(leadingZeroBytes * 2, leadingZeroBytes * 2 + 2) === "00") {
        leadingZeroBytes += 1;
      }

      let number = BigInt(`0x${hex}`);
      let encoded = "";
      while (number > 0n) {
        encoded = URL_ID_ALPHABET[Number(number % 58n)] + encoded;
        number /= 58n;
      }

      if (!encoded && leadingZeroBytes === 0) {
        encoded = URL_ID_ALPHABET[0];
      }

      return URL_ID_ALPHABET[0].repeat(leadingZeroBytes) + encoded;
    }

    function decodeUrlId(value) {
      const text = sanitizeText(value, 120).trim();
      if (!text) return "";

      let number = 0n;
      for (const char of text) {
        const digit = URL_ID_DIGITS[char];
        if (digit === undefined) return "";
        number = number * 58n + BigInt(digit);
      }

      let hex = number.toString(16);
      if (hex.length % 2 === 1) {
        hex = `0${hex}`;
      }

      let leadingZeroBytes = 0;
      while (text[leadingZeroBytes] === URL_ID_ALPHABET[0]) {
        leadingZeroBytes += 1;
      }
      if (leadingZeroBytes > 0) {
        hex = `${"00".repeat(leadingZeroBytes)}${hex}`;
      }

      if (hex.length > 32) return "";
      const normalized = hex.padStart(32, "0");
      return isInternalId(normalized) ? normalized : "";
    }

    function sanitizeText(value, maxLength) {
      if (value === null || value === undefined) return "";
      return String(value).trim().slice(0, maxLength);
    }

    function sanitizeUploadedImageUrl(value) {
      const text = sanitizeText(value, 2000);
      if (!text) return "";
      try {
        const url = new URL(text);
        if (url.protocol !== "https:") return "";
        if (url.hostname !== "user.uploads.dev") return "";
        if (!url.pathname.startsWith("/file/")) return "";
        return url.href;
      } catch (error) {
        return "";
      }
    }

    function sanitizeCommentsPluginUsername(value) {
      const text = sanitizeText(value, 60);
      return /^[a-zA-Z0-9._-]+$/.test(text) ? text : "";
    }

    function sanitizeMessageDingVolume(value) {
      const number = Number(value);
      if (!Number.isFinite(number)) return 0.35;
      return Math.min(1, Math.max(0, number));
    }

    function sanitizeThemeId(value) {
      const text = sanitizeText(value, 40).toLowerCase();
      return PROFILE_THEME_IDS.includes(text) ? text : PROFILE_THEME_IDS[0];
    }

    function sanitizeProfile(input) {
      const source = input && typeof input === "object" ? input : {};
      return {
        name: sanitizeText(source.name, 80),
        description: sanitizeText(source.description, 4000),
        blogTitle: sanitizeText(source.blogTitle, 80),
        profileImageUrl: sanitizeUploadedImageUrl(source.profileImageUrl),
        backgroundImageUrl: sanitizeUploadedImageUrl(source.backgroundImageUrl),
        commentsPluginUsername: sanitizeCommentsPluginUsername(source.commentsPluginUsername),
        themeId: sanitizeThemeId(source.themeId),
      };
    }

    function isPublicKey(text) {
      return typeof text === "string" && text.startsWith("PUBLIC_") && text.endsWith("_PUBLIC_END");
    }

    function isPrivateKey(text) {
      return typeof text === "string" && text.startsWith("PRIVATE_") && text.endsWith("_PRIVATE_END");
    }

    function emptyStorage() {
      return {
        primaryInboxId: "",
        inboxes: {},
        blockedUsers: {},
      };
    }

    function sanitizeStorage(input) {
      const base = emptyStorage();
      if (!input || typeof input !== "object") return base;

      const primaryInboxId = normalizeId(input.primaryInboxId);
      const inboxes = {};
      const blockedUsers = {};

      if (input.inboxes && typeof input.inboxes === "object") {
        for (const key of Object.keys(input.inboxes)) {
          const inboxId = normalizeId(key);
          const entry = input.inboxes[key];
          if (!inboxId || !entry || typeof entry !== "object") continue;
          if (!isInternalId(inboxId)) continue;
          if (!isPublicKey(entry.publicKey) || !isPrivateKey(entry.privateKey)) continue;
          inboxes[inboxId] = {
            inboxId,
            publicKey: entry.publicKey,
            privateKey: entry.privateKey,
            uploadUrl: typeof entry.uploadUrl === "string" ? entry.uploadUrl : "",
            readerLink: typeof entry.readerLink === "string" ? entry.readerLink.trim() : "",
            playMessageDing: entry.playMessageDing === true,
            playMessageDingVolume: sanitizeMessageDingVolume(entry.playMessageDingVolume),
            role: entry.role === "reader" ? "reader" : "owner",
            profile: sanitizeProfile(entry.profile),
            createdAt: typeof entry.createdAt === "number" ? entry.createdAt : Date.now(),
          };
        }
      }

      if (input.blockedUsers && typeof input.blockedUsers === "object") {
        for (const key of Object.keys(input.blockedUsers)) {
          const inboxId = normalizeId(key);
          const list = input.blockedUsers[key];
          if (!inboxId || !list || typeof list !== "object") continue;
          if (!isInternalId(inboxId)) continue;
          blockedUsers[inboxId] = {};
          for (const userIdKey of Object.keys(list)) {
            const item = list[userIdKey];
            const userId = typeof item?.userId === "string" ? item.userId : typeof userIdKey === "string" ? userIdKey : "";
            if (!userId) continue;
            const label = typeof item?.label === "string" ? item.label.replace(/[\r\n]+/g, " ").trim().slice(0, 32) : "";
            blockedUsers[inboxId][userId] = {
              userId,
              label: label || "Blocked message",
            };
          }
        }
      }

      const resolvedPrimary = primaryInboxId && inboxes[primaryInboxId] ? primaryInboxId : "";

      return {
        primaryInboxId: resolvedPrimary,
        inboxes,
        blockedUsers,
      };
    }

    function loadStorage() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (!raw) return emptyStorage();
        return sanitizeStorage(JSON.parse(raw));
      } catch (error) {
        console.error("Failed to read profile storage.", error);
        return emptyStorage();
      }
    }

    function saveStorage(storage) {
      const clean = sanitizeStorage(storage);
      localStorage.setItem(STORAGE_KEY, JSON.stringify(clean));
      return clean;
    }

    function parseHashValue() {
      const raw = (window.location.hash || "").slice(1).trim();
      return raw;
    }

    function loadBlogSeed() {
      try {
        const raw = sessionStorage.getItem(BLOG_SEED_KEY);
        if (!raw) return { inboxId: "", commentsPluginUsername: "" };
        const parsed = JSON.parse(raw);
        const inboxId = normalizeId(parsed?.inboxId);
        const commentsPluginUsername = sanitizeCommentsPluginUsername(parsed?.commentsPluginUsername);
        if (!inboxId || !commentsPluginUsername) return { inboxId: "", commentsPluginUsername: "" };
        return { inboxId, commentsPluginUsername };
      } catch (error) {
        return { inboxId: "", commentsPluginUsername: "" };
      }
    }

    const storage = loadStorage();
    const blogSeed = loadBlogSeed();
    const rawHash = parseHashValue();
    const isHomepageRoute = !rawHash || rawHash.toLowerCase() === "edit";
    const normalizedHash = isHomepageRoute ? "" : rawHash;
    const hashParts = normalizedHash ? normalizedHash.split("~") : [];
    const hasAccessSegment = hashParts.length > 1 && Boolean(hashParts[1]);
    let inboxId = "";
    let accessId = "";
    let invalidRoute = false;

    if (!isHomepageRoute) {
      if (hashParts.length < 1 || hashParts.length > 2 || !hashParts[0]) {
        invalidRoute = true;
      } else {
        inboxId = decodeUrlId(hashParts[0]);
        accessId = hashParts[1] ? decodeUrlId(hashParts[1]) : "";
        invalidRoute = !inboxId || (Boolean(hashParts[1]) && !accessId);
      }
    }

    if (isHomepageRoute && storage.primaryInboxId) {
      inboxId = normalizeId(storage.primaryInboxId);
    }

    const cachedCommentsPluginUsername = sanitizeCommentsPluginUsername(storage.inboxes[inboxId]?.profile?.commentsPluginUsername);
    const initialCommentsPluginUsername = inboxId
      ? cachedCommentsPluginUsername || (blogSeed.inboxId === inboxId ? blogSeed.commentsPluginUsername : "")
      : "";
    const initialInboxChannel = buildInboxChannel(inboxId);
    const initialBlogChannel = buildBlogChannel(inboxId, initialCommentsPluginUsername);
    const initialOwnsInbox = Boolean(storage.inboxes[inboxId]?.privateKey);

    window.secretInboxLoadStorage = loadStorage;
    window.secretInboxSaveStorage = saveStorage;
    window.secretInboxBlogSeedKey = BLOG_SEED_KEY;
    window.secretInboxEncodeUrlId = encodeUrlId;
    window.secretInboxBuildInboxChannel = buildInboxChannel;
    window.secretInboxBuildBlogChannel = buildBlogChannel;
    window.secretInboxThemeIds = PROFILE_THEME_IDS.slice();
    window.secretInboxInitialInboxChannel = initialInboxChannel;
    window.secretInboxInitialBlog = {
      inboxId,
      commentsPluginUsername: initialCommentsPluginUsername,
      channel: initialBlogChannel,
      ownsInbox: initialOwnsInbox,
    };
    window.secretInboxCommentsLoaded = false;
    window.secretInboxCommentsLoadedAt = 0;
    window.secretInboxBlogLoaded = false;
    window.secretInboxBlogLoadedAt = 0;
    window.secretInboxRoute = {
      isHomepageRoute,
      invalidRoute,
      hasAccessSegment,
      inboxId,
      accessId,
    };
  })();

  window.secretInboxPreferredColorScheme = window.matchMedia && window.matchMedia("(prefers-color-scheme: dark)").matches ? "dark" : "light";

  window.commentOptions = {

    function normalizeId(value) {
      return String(value || "").trim().toLowerCase().replace(/[^0-9a-f]/g, "");
    }

    function isInternalId(value) {
      return /^[0-9a-f]{32}$/.test(String(value || ""));
    }

    function sanitizeText(value, maxLength) {
      if (value === null || value === undefined) return "";
      return String(value).trim().slice(0, maxLength);
    }

    function sanitizeUploadedImageUrl(value) {
      const text = sanitizeText(value, 2000);
      if (!text) return "";
      try {
        const url = new URL(text);
        if (url.protocol !== "https:") return "";
        if (url.hostname !== "user.uploads.dev") return "";
        if (!url.pathname.startsWith("/file/")) return "";
        return url.href;
      } catch (error) {
        return "";
      }
    }

    function sanitizeCommentsPluginUsername(value) {
      const text = sanitizeText(value, 60);
      return /^[a-zA-Z0-9._-]+$/.test(text) ? text : "";
    }

    function sanitizeMessageDingVolume(value) {
      const number = Number(value);
      if (!Number.isFinite(number)) return 0.35;
      return Math.min(1, Math.max(0, number));
    }

    function sanitizeThemeId(value) {
      const text = sanitizeText(value, 40).toLowerCase();
      return PROFILE_THEME_IDS.includes(text) ? text : PROFILE_THEME_IDS[0];
    }

    function sanitizeProfile(value) {
      const source = value && typeof value === "object" ? value : {};
      return {
        name: sanitizeText(source.name, 80),
        description: sanitizeText(source.description, 4000),
        blogTitle: sanitizeText(source.blogTitle, 80),
        profileImageUrl: sanitizeUploadedImageUrl(source.profileImageUrl),
        backgroundImageUrl: sanitizeUploadedImageUrl(source.backgroundImageUrl),
        commentsPluginUsername: sanitizeCommentsPluginUsername(source.commentsPluginUsername),
        themeId: sanitizeThemeId(source.themeId),
      };
    }

    function isPublicKey(text) {
      return typeof text === "string" && text.startsWith("PUBLIC_") && text.endsWith("_PUBLIC_END");
    }

    function isPrivateKey(text) {
      return typeof text === "string" && text.startsWith("PRIVATE_") && text.endsWith("_PRIVATE_END");
    }

    function isEncryptedMessage(text) {
      return typeof text === "string" && /^ENCRYPTED_[0-9]+_[\s\S]+_ENCRYPTED_END$/.test(text);
    }

    function doesKeyPairMatch(publicKey, privateKey) {
      if (!isPublicKey(publicKey) || !isPrivateKey(privateKey)) return false;
      try {
        const probe = `probe:${Date.now()}:${Math.random().toString(36).slice(2)}`;
        return secret.decrypt(secret.encrypt(probe, publicKey), privateKey) === probe;
      } catch (error) {
        return false;
      }
    }

    function getPreferredThemeMode() {
      return window.matchMedia && window.matchMedia("(prefers-color-scheme: dark)").matches ? "dark" : "light";
    }

    function getContrastTextColor(hexColor) {
      const text = sanitizeText(hexColor, 32);
      const match = text.match(/^#([0-9a-f]{6})$/i);
      if (!match) return "#ffffff";
      const hex = match[1];
      const rgb = [0, 2, 4].map((index) => parseInt(hex.slice(index, index + 2), 16) / 255);
      const [r, g, b] = rgb.map((channel) => (
        channel <= 0.03928 ? channel / 12.92 : ((channel + 0.055) / 1.055) ** 2.4
      ));
      const luminance = 0.2126 * r + 0.7152 * g + 0.0722 * b;
      const whiteContrast = 1.05 / (luminance + 0.05);
      const blackContrast = (luminance + 0.05) / 0.05;
      return blackContrast >= whiteContrast ? "#111111" : "#ffffff";
    }

    function applyThemeColors(themeId) {
      const selectedThemeId = sanitizeThemeId(themeId);
      const theme = PROFILE_THEMES[selectedThemeId] || PROFILE_THEMES[PROFILE_THEME_IDS[0]];

      const accentText = getContrastTextColor(palette.accent);
      const accentHoverText = getContrastTextColor(palette.accentHover);

      rootStyle.setProperty("--accent-hover", palette.accentHover);
      rootStyle.setProperty("--accent-text", accentText);
      rootStyle.setProperty("--accent-hover-text", accentHoverText);

    async function preloadProfileImage(url) {
      const response = await fetch(url, { cache: "force-cache" });
      if (!response.ok) {
        throw new Error(`Image request failed with status ${response.status}.`);
      }
      const blob = await response.blob();
      return URL.createObjectURL(blob);
    }

    async function fetchImageInfo(url) {
      const response = await fetch(`https://upload.perchance.org/api/fileInfo?url=${encodeURIComponent(url)}`, { cache: "no-store" });
      if (!response.ok) {
        throw new Error(`Image info request failed with status ${response.status}.`);
      }
      const result = await response.json();
      if (!result || result.error) {
        throw new Error(result?.error || "Image info request failed.");
      }
      return result;
    }

    async function resolveProfileImageAsset(url) {
      const safeUrl = sanitizeUploadedImageUrl(url);
      if (!safeUrl) {
        return {
          sourceUrl: "",
          displayUrl: "",
          shouldHide: false,
          hasSensitiveTag: false,
        };
      }

      if (state.profileMediaCache.has(safeUrl)) {
        return state.profileMediaCache.get(safeUrl);
      }

      const promise = (async () => {
        const [infoResult, imageResult] = await Promise.allSettled([
          fetchImageInfo(safeUrl),
          preloadProfileImage(safeUrl),
        ]);

        const hasSensitiveTag = infoResult.status === "fulfilled" && Array.isArray(infoResult.value?.tags) && infoResult.value.tags.includes("nsfw");
        const shouldHide = infoResult.status !== "fulfilled" || hasSensitiveTag;
        const displayUrl = imageResult.status === "fulfilled" ? imageResult.value : safeUrl;

        return {
          sourceUrl: safeUrl,
          displayUrl,
          shouldHide,
          hasSensitiveTag,
        };
      })();

      state.profileMediaCache.set(safeUrl, promise);
      return promise;
    }

    async function resolveProfileMediaState(profile) {
      const safeProfile = sanitizeProfile(profile || {});
      const token = ++state.profileMediaResolveToken;
      clearResolvedProfileMediaUrls();
      state.profileMediaState = {
        status: "loading",
        requestedProfile: safeProfile,
        profileImageUrl: safeProfile.profileImageUrl,
        backgroundImageUrl: safeProfile.backgroundImageUrl,
        profileImageObjectUrl: "",
        backgroundImageObjectUrl: "",
        hideSensitiveContent: false,
        revealSensitiveContent: false,
      };

      const [profileImageAsset, backgroundImageAsset] = await Promise.all([
        resolveProfileImageAsset(safeProfile.profileImageUrl),
        resolveProfileImageAsset(safeProfile.backgroundImageUrl),
      ]);

      if (token !== state.profileMediaResolveToken) return;

      state.profileMediaState = {
        status: "ready",
        requestedProfile: safeProfile,
        profileImageUrl: safeProfile.profileImageUrl,
        backgroundImageUrl: safeProfile.backgroundImageUrl,
        profileImageObjectUrl: profileImageAsset.displayUrl !== profileImageAsset.sourceUrl ? profileImageAsset.displayUrl : "",
        backgroundImageObjectUrl: backgroundImageAsset.displayUrl !== backgroundImageAsset.sourceUrl ? backgroundImageAsset.displayUrl : "",
        hideSensitiveContent: profileImageAsset.shouldHide || backgroundImageAsset.shouldHide,
        revealSensitiveContent: false,
      };
    }

    function getResolvedProfileForDisplay(profile) {
      const safeProfile = sanitizeProfile(profile || {});
      const mediaState = state.profileMediaState;
      const mediaMatchesProfile = mediaState.status === "ready"
        && mediaState.requestedProfile
        && mediaState.requestedProfile.profileImageUrl === safeProfile.profileImageUrl
        && mediaState.requestedProfile.backgroundImageUrl === safeProfile.backgroundImageUrl;

      if (!mediaMatchesProfile) {
        return {
          profile: {
            ...safeProfile,
            profileImageUrl: "",
            backgroundImageUrl: "",
          },
          hideSensitiveContent: false,
          mediaReady: !safeProfile.profileImageUrl && !safeProfile.backgroundImageUrl,
        };
      }

      if (mediaState.hideSensitiveContent && !mediaState.revealSensitiveContent) {
        return {
          profile: {
            ...safeProfile,
            description: "",
            profileImageUrl: "",
            backgroundImageUrl: "",
          },
          hideSensitiveContent: true,
          mediaReady: true,
        };
      }

      return {
        profile: {
          ...safeProfile,
          profileImageUrl: mediaState.profileImageObjectUrl || safeProfile.profileImageUrl,
          backgroundImageUrl: mediaState.backgroundImageObjectUrl || safeProfile.backgroundImageUrl,
        },
        hideSensitiveContent: false,
        mediaReady: true,
      };
    }

      if (!inboxId) {
        window.location.hash = "";
      } else {
        const urlInboxId = window.secretInboxEncodeUrlId(inboxId);
        const urlAccessId = accessId ? window.secretInboxEncodeUrlId(accessId) : "";
        window.location.hash = urlAccessId ? `${urlInboxId}~${urlAccessId}` : urlInboxId;
      }
      window.location.reload();
    }

    function parseUploadId(uploadUrl) {
      try {
        const pathname = new URL(uploadUrl, window.location.href).pathname;
        const filename = pathname.split("/").filter(Boolean).pop() || "";
        const inboxId = normalizeId(filename.split(".")[0] || "");
        return isInternalId(inboxId) ? inboxId : "";
      } catch (error) {
        console.error("Failed to parse upload URL.", error);
        return "";
      }
    }

    function getUploadFetchUrls(fileId, exactUrl) {
      const urls = [];
      if (exactUrl) urls.push(exactUrl);
      urls.push(`https://user.uploads.dev/file/${fileId}.txt`);
      urls.push(`https://user.uploads.dev/file/${fileId}`);
      return [...new Set(urls)];
    }

    async function fetchTextFromUpload(fileId, exactUrl) {
      const urls = getUploadFetchUrls(fileId, exactUrl);
      for (const url of urls) {
        try {
          const response = await fetch(url, { cache: "no-store" });
          if (!response.ok) continue;
          return (await response.text()).trim();
        } catch (error) {
          console.warn("Failed to fetch uploaded file from", url, error);
        }
      }
      throw new Error("The uploaded file could not be loaded.");
    }

    async function fetchInboxBundle(inboxId, uploadUrl) {
      const text = await fetchTextFromUpload(inboxId, uploadUrl);
      let parsed;
      try {
        parsed = JSON.parse(text);
      } catch (error) {
        throw new Error("This profile link is invalid or unavailable.");
      }
      if (!isPublicKey(parsed?.publicKey)) {
        throw new Error("This profile link is invalid or unavailable.");
      }
      return {
        publicKey: parsed.publicKey,
        profile: sanitizeProfile(parsed.profile),
      };
    }

    async function fetchReaderBundle(accessId) {
      const text = await fetchTextFromUpload(accessId);
      let parsed;
      try {
        parsed = JSON.parse(text);
      } catch (error) {
        throw new Error("This profile access link is invalid.");
      }

      const inboxId = normalizeId(parsed?.inboxId);
      const publicKey = parsed?.publicKey;
      const privateKey = parsed?.privateKey;

      if (!isInternalId(inboxId) || !isPublicKey(publicKey) || !isPrivateKey(privateKey)) {
        throw new Error("This profile access link is invalid.");
      }

      return {
        inboxId,
        publicKey,
        privateKey,
      };
    }

    function upsertInbox(entry) {
      const inboxId = normalizeId(entry.inboxId);
      if (!isInternalId(inboxId) || !isPublicKey(entry.publicKey) || !isPrivateKey(entry.privateKey)) return;

      const existing = state.storage.inboxes[inboxId] || {};
      state.storage.inboxes[inboxId] = {
        inboxId,
        publicKey: entry.publicKey,
        privateKey: entry.privateKey,
        uploadUrl: entry.uploadUrl || existing.uploadUrl || "",
        readerLink: entry.readerLink || existing.readerLink || "",
        playMessageDing: entry.playMessageDing === true || (entry.playMessageDing !== false && existing.playMessageDing === true),
        playMessageDingVolume: sanitizeMessageDingVolume(entry.playMessageDingVolume ?? existing.playMessageDingVolume),
        role: existing.role === "owner" || entry.role === "owner" ? "owner" : "reader",
        profile: sanitizeProfile(entry.profile || existing.profile),
        createdAt: existing.createdAt || entry.createdAt || Date.now(),
      };

      if (!state.storage.primaryInboxId) {
        state.storage.primaryInboxId = inboxId;
      }
    }

    async function ensureAudioPlaybackReady() {
      const audioContext = getMessageDingAudioContext();
      if (!audioContext) {
        state.audioNeedsUserGesture = false;
        updateAudioEnableButton();
        return false;
      }

      try {
        if (audioContext.state === "suspended") {
          await audioContext.resume();
        }
      } catch (error) {
        state.audioNeedsUserGesture = true;
        updateAudioEnableButton();
        return false;
      }

      state.audioNeedsUserGesture = audioContext.state !== "running";
      updateAudioEnableButton();
      return audioContext.state === "running";
    }

    async function playDingTone(volume, options = {}) {
      const audioContext = getMessageDingAudioContext();
      if (!audioContext || audioContext.state !== "running") return false;

      const start = audioContext.currentTime;
      const oscillator = audioContext.createOscillator();
      const gainNode = audioContext.createGain();
      const safeVolume = Math.max(0.0001, sanitizeMessageDingVolume(volume));

      oscillator.type = "sine";
      oscillator.frequency.setValueAtTime(900, start);
      oscillator.frequency.exponentialRampToValueAtTime(1320, start + 0.16);

      gainNode.gain.setValueAtTime(0.0001, start);
      gainNode.gain.exponentialRampToValueAtTime(safeVolume * 0.4, start + 0.02);
      gainNode.gain.exponentialRampToValueAtTime(0.0001, start + 0.24);

      oscillator.connect(gainNode);
      gainNode.connect(audioContext.destination);
      oscillator.start(start);
      oscillator.stop(start + 0.26);
      if (options.markAsPlayed !== false) {
        state.lastMessageDingAt = Date.now();
      }
      return true;
    }

    async function probeAudioPlayback() {
      const ready = await ensureAudioPlaybackReady();
      if (!ready) return false;
      await playDingTone(0.0001, { markAsPlayed: false });
      return true;

    async function playMessageDing(options = {}) {
      const currentEntry = getCurrentInboxEntry();
      if (!options.force && !currentEntry?.playMessageDing) return;
      const now = Date.now();
      if (!options.force && now - state.lastMessageDingAt < 700) return;

      const ready = await ensureAudioPlaybackReady();
      if (!ready) return;
      await playDingTone(options.volume ?? getCurrentMessageDingVolume(), {
        markAsPlayed: options.markAsPlayed,
      });
    }

    async function uploadCreateImage(file, label) {

    async function createProfileAndReload() {
      const uploadedImages = {
        backgroundImageUrl: await uploadCreateImage(state.createBackgroundImageFile, "Background image"),
        profileImageUrl: await uploadCreateImage(state.createProfileImageFile, "Profile picture"),
      };
      const keys = secret.generateKeyPair();
      const profile = collectCreateProfile(uploadedImages);
      const uploadResult = await upload(JSON.stringify({
        publicKey: keys.public,
        profile,
      }));

    function appendTextWithLinks(container, text, options = {}) {
      const allowPerchance = options.allowPerchance !== false;
      const allowUploads = options.allowUploads === true;
      const stripPerchanceProtocol = options.stripPerchanceProtocol === true;
      const parts = [];
      if (allowPerchance) parts.push(String.raw`\b(?:https?:\/\/)?perchance\.org\/[^\s]+`);
      if (allowUploads) parts.push(String.raw`\bhttps:\/\/user\.uploads\.dev\/file\/[^\s]+`);
      const pattern = parts.length ? new RegExp(parts.join("|"), "gi") : null;
      const lines = String(text || "").split("\n");

      lines.forEach((line, lineIndex) => {
        let lastIndex = 0;
        if (pattern) {
          line.replace(pattern, (match, offset) => {
            if (offset > lastIndex) {
              container.appendChild(document.createTextNode(line.slice(lastIndex, offset)));
            }

            const href = /^https?:\/\//i.test(match) ? match : `https://${match}`;
            const link = document.createElement("a");
            link.href = href;
            link.target = "_blank";
            link.rel = "noopener noreferrer";
            link.textContent = stripPerchanceProtocol && /perchance\.org\//i.test(match) ? match.replace(/^https?:\/\//i, "") : match;
            container.appendChild(link);
            lastIndex = offset + match.length;
            return match;
          });
        }

        if (lastIndex < line.length) {
          container.appendChild(document.createTextNode(line.slice(lastIndex)));
        }

        if (lineIndex < lines.length - 1) {
          container.appendChild(document.createElement("br"));
        }
      });
    }

          scheduleBlogEmptyStateRefresh();

        els.blogLoadingState.hidden = true;
        els.blogEmptyState.hidden = false;
        els.blogFeed.appendChild(els.blogEmptyState);
        return;

        state.audioNeedsUserGesture = false;
        updateAudioEnableButton();
        return;
      }
      els.newProfileSection.hidden = false;
      els.accessToolsSection.hidden = !state.currentCanRead;
      els.notificationsSection.hidden = !state.currentCanRead;
      if (!state.currentCanRead) {
        els.readerLinkStatus.textContent = "";
        els.readerLinkWrap.hidden = true;
        els.createReaderLinkButton.hidden = false;
        els.playMessageDingCheckbox.checked = false;
        els.playMessageDingVolumeInput.value = "35";
        updateMessageDingVolumeLabel();
        state.audioNeedsUserGesture = false;
        updateAudioEnableButton();
      } else {
        const currentEntry = getCurrentInboxEntry() || {};
        const savedReaderLink = currentEntry.readerLink || "";
        if (savedReaderLink) {
          els.readerLinkInput.value = stripShareUrlProtocol(savedReaderLink);
          els.readerLinkWrap.hidden = false;
          els.createReaderLinkButton.hidden = true;
        } else {
          els.readerLinkInput.value = "";
          els.readerLinkWrap.hidden = true;
          els.createReaderLinkButton.hidden = false;
        }
        els.playMessageDingCheckbox.checked = currentEntry.playMessageDing === true;
        els.playMessageDingVolumeInput.value = String(Math.round(sanitizeMessageDingVolume(currentEntry.playMessageDingVolume) * 100));
        updateMessageDingVolumeLabel();
        if (!currentEntry.playMessageDing) {
          state.audioNeedsUserGesture = false;
          updateAudioEnableButton();
        }
      }
      renderSavedInboxes();
      renderBlockedUsers();
    }

    function getMessageKey(message) {
      return `${message.sortTime}|${message.timestamp}|${message.message}`;
    }

    function getMessagesTopThreshold() {
      const firstCard = els.messagesFeed.querySelector(".message-card");
      const messageHeight = firstCard ? firstCard.offsetHeight : 88;
      return Math.max(messageHeight * 2, 140);
    }

    function isMessagesFeedNearTop() {
      return els.messagesFeed.scrollTop <= getMessagesTopThreshold();
    }

    function scrollMessagesToTop(behavior = "auto") {
      els.messagesFeed.scrollTo({
        top: 0,
        behavior,
      });
      state.hasUnreadMessagesBelow = false;
      els.jumpToNewMessagesButton.hidden = true;
    }

    function updateNewMessagesIndicator() {
      if (isMessagesFeedNearTop()) {
        state.hasUnreadMessagesBelow = false;
      }
      els.jumpToNewMessagesButton.hidden = !state.hasUnreadMessagesBelow;
    }

    function clearMessagesEmptyStateTimer() {
      if (state.messagesEmptyStateTimer) {
        window.clearTimeout(state.messagesEmptyStateTimer);
        state.messagesEmptyStateTimer = null;
      }
    }

    function scheduleMessagesEmptyStateRefresh() {
      if (!window.secretInboxCommentsLoaded) {
        clearMessagesEmptyStateTimer();
        return;
      }

      const loadedAt = Number(window.secretInboxCommentsLoadedAt || 0);
      const remainingDelay = MESSAGE_EMPTY_STATE_DELAY_MS - (Date.now() - loadedAt);
      if (remainingDelay <= 0) {
        clearMessagesEmptyStateTimer();
        return;
      }

      clearMessagesEmptyStateTimer();
      state.messagesEmptyStateTimer = window.setTimeout(() => {
        state.messagesEmptyStateTimer = null;
        renderInbox();
      }, remainingDelay + 20);
    }

    function clearBlogEmptyStateTimer() {
      if (state.blogEmptyStateTimer) {
        window.clearTimeout(state.blogEmptyStateTimer);
        state.blogEmptyStateTimer = null;
      }
    }

    function scheduleBlogEmptyStateRefresh() {
      if (!window.secretInboxBlogLoaded) {
        clearBlogEmptyStateTimer();
        return;
      }

      const loadedAt = Number(window.secretInboxBlogLoadedAt || 0);
      const remainingDelay = MESSAGE_EMPTY_STATE_DELAY_MS - (Date.now() - loadedAt);
      if (remainingDelay <= 0) {
        clearBlogEmptyStateTimer();
        return;
      }

      clearBlogEmptyStateTimer();
      state.blogEmptyStateTimer = window.setTimeout(() => {
        state.blogEmptyStateTimer = null;
        renderBlog();
      }, remainingDelay + 20);
    }

    function buildBlockedMessagePreview(text) {
      const preview = String(text || "").replace(/[\r\n]+/g, " ").trim().slice(0, 32);
      return preview || "Blocked message";
    }

    function blockSender(message) {
      if (!state.currentInbox || !message.userId) return;
      const confirmed = window.confirm(`Block this sender?\n\nTheir messages will be hidden on this device. You can 𝘂𝗻𝗯𝗹𝗼𝗰𝗸 them in the 𝗦𝗲𝘁𝘁𝗶𝗻𝗴𝘀 panel.`);
      if (!confirmed) return;

      getBlockedMap(state.currentInbox.inboxId)[message.userId] = {
        userId: message.userId,
        label: buildBlockedMessagePreview(message.message),
      };
      saveStorage(state.storage);
      renderAdvanced();
      renderInbox();
    }

    function buildMessageCard(message) {
      const article = document.createElement("article");
      article.className = "message-card";

      const header = document.createElement("div");
      header.className = "message-meta";

      const left = document.createElement("div");
      left.className = "message-meta-main";

      if (message.timestamp) {
        const time = document.createElement("time");
        time.className = "message-time";
        time.textContent = message.timestamp;
        left.appendChild(time);
      }

      header.appendChild(left);

      if (state.currentOwnsInbox && message.userId) {
        const blockButton = document.createElement("button");
        blockButton.type = "button";
        blockButton.className = "danger-button message-block-button";
        blockButton.textContent = "Block";
        blockButton.addEventListener("click", () => {
          blockSender(message);
        });
        header.appendChild(blockButton);
      }

      const body = document.createElement("p");
      body.className = "message-body";
      body.textContent = message.message;

      article.appendChild(header);
      article.appendChild(body);
      return article;
    }

    function getReadableMessages() {
      if (!state.currentCanRead || !state.currentPrivateKey) return [];
      const comments = Array.isArray(window.secretInboxComments) ? window.secretInboxComments : [];
      const blocked = state.currentInbox ? getBlockedMap(state.currentInbox.inboxId) : {};
      const messages = [];

      for (const comment of comments) {
        if (!comment || typeof comment.message !== "string") continue;
        if (!isEncryptedMessage(comment.message.trim())) continue;

        const userId = typeof comment?.user?.id === "string" ? comment.user.id : "";
        if (userId && blocked[userId]) continue;

        try {
          const plaintext = secret.decrypt(comment.message.trim(), state.currentPrivateKey);
          messages.push({
            timestamp: buildTimestamp(comment),
            sortTime: Number(getCommentTime(comment) || 0),
            message: plaintext,
            userId,
          });
        } catch (error) {
          continue;
        }
      }

      messages.sort((a, b) => b.sortTime - a.sortTime);
      return messages;
    }

    function renderInbox() {
      if (!state.currentCanRead || !state.initComplete) return;
      const messages = getReadableMessages();
      const nextKeys = messages.map(getMessageKey);
      const previousKeys = state.lastRenderedMessageKeys;
      const previousScrollHeight = els.messagesFeed.scrollHeight;
      const previousScrollTop = els.messagesFeed.scrollTop;
      const wasNearTop = isMessagesFeedNearTop();
      const prependedNewMessages = previousKeys.length > 0
        && nextKeys.length > previousKeys.length
        && previousKeys.every((key, index) => nextKeys[index + (nextKeys.length - previousKeys.length)] === key);

      els.messagesFeed.replaceChildren();
      state.lastRenderedMessageKeys = nextKeys;

      if (!window.secretInboxCommentsLoaded) {
        clearMessagesEmptyStateTimer();
        els.messagesLoadingState.hidden = false;
        els.emptyState.hidden = true;
        els.messagesFeed.appendChild(els.messagesLoadingState);
        state.hasUnreadMessagesBelow = false;
        els.jumpToNewMessagesButton.hidden = true;
        return;
      }

      if (!messages.length) {
        const loadedAt = Number(window.secretInboxCommentsLoadedAt || 0);
        if (!loadedAt || Date.now() - loadedAt < MESSAGE_EMPTY_STATE_DELAY_MS) {
          scheduleMessagesEmptyStateRefresh();
          els.messagesLoadingState.hidden = false;
          els.emptyState.hidden = true;
          els.messagesFeed.appendChild(els.messagesLoadingState);
          state.hasUnreadMessagesBelow = false;
          els.jumpToNewMessagesButton.hidden = true;
          return;
        }
      }

      if (!messages.length) {
        clearMessagesEmptyStateTimer();
        els.messagesLoadingState.hidden = true;
        els.emptyState.hidden = false;
        els.messagesFeed.appendChild(els.emptyState);
        state.hasUnreadMessagesBelow = false;
        els.jumpToNewMessagesButton.hidden = true;
        return;
      }

      clearMessagesEmptyStateTimer();
      els.messagesLoadingState.hidden = true;
      els.emptyState.hidden = true;
      for (const message of messages) {
        els.messagesFeed.appendChild(buildMessageCard(message));
      }

      if (!previousKeys.length || wasNearTop) {
        scrollMessagesToTop();
        return;
      }

      if (prependedNewMessages) {
        const scrollHeightDelta = els.messagesFeed.scrollHeight - previousScrollHeight;
        els.messagesFeed.scrollTop = previousScrollTop + scrollHeightDelta;
        state.hasUnreadMessagesBelow = true;
        updateNewMessagesIndicator();
        return;
      }

      els.messagesFeed.scrollTop = previousScrollTop;
      updateNewMessagesIndicator();
    }

    async function copyInputValue(inputEl, buttonEl, successText) {
      const value = inputEl.value.trim();
      if (!value) return;
      await navigator.clipboard.writeText(value);
      flashButtonText(buttonEl, successText);
    }

    function flashButtonText(buttonEl, successText, duration = 1600) {
      const original = buttonEl.textContent;
      const lockedWidth = `${Math.ceil(buttonEl.getBoundingClientRect().width)}px`;
      buttonEl.style.width = lockedWidth;
      buttonEl.textContent = successText;
      setTimeout(() => {
        buttonEl.textContent = original;
        buttonEl.style.width = "";
      }, duration);
    }

    async function sendMessage() {
      const plaintext = els.composeInput.value.trim();
      if (!plaintext) {
        els.composeStatus.textContent = "Write a message first.";
        return;
      }

      els.sendButton.disabled = true;
      els.composeStatus.textContent = "Sending...";

      try {
        const controller = await waitForCommentsController();
        const encryptedMessage = secret.encrypt(plaintext, state.currentPublicKey);
        if (typeof controller.setNicknameForNextComment === "function") {
          await controller.setNicknameForNextComment("");
        }
        await controller.submit(encryptedMessage);
        els.composeInput.value = "";
        els.composeStatus.textContent = "Sent.";
      } catch (error) {
        console.error("Failed to send secret message.", error);
        const message = typeof error === "string" ? error : error?.message;
        els.composeStatus.textContent = message || "Failed to send message.";
      } finally {
        els.sendButton.disabled = false;
      }
    }

    async function createReadLink() {
      if (!state.currentInbox || !state.currentCanRead) return;

      els.createReaderLinkButton.disabled = true;
      els.readerLinkStatus.textContent = "Creating profile access link...";
      els.readerLinkWrap.hidden = true;

      try {
        const payload = JSON.stringify({
          inboxId: state.currentInbox.inboxId,
          publicKey: state.currentPublicKey,
          privateKey: state.currentPrivateKey,
        });
        const uploadResult = await upload(payload);
        if (uploadResult?.error || !uploadResult?.url) {
          throw new Error(uploadResult?.error || "Failed to create the profile access link.");
        }

        const accessId = parseUploadId(uploadResult.url);
        if (!accessId) {
          throw new Error("The profile access link could not be created.");
        }

        const readerLink = stripShareUrlProtocol(buildReadUrl(state.currentInbox.inboxId, accessId));
        els.readerLinkInput.value = readerLink;
        els.readerLinkWrap.hidden = false;
        els.createReaderLinkButton.hidden = true;
        els.readerLinkStatus.textContent = "Read link created.";
        const currentEntry = getCurrentInboxEntry() || {};
        upsertInbox({
          ...currentEntry,
          inboxId: state.currentInbox.inboxId,
          publicKey: state.currentPublicKey,
          privateKey: state.currentPrivateKey,
          uploadUrl: state.currentInbox.uploadUrl || "",
          role: state.currentInbox.role === "owner" ? "owner" : "reader",
          profile: state.currentProfile || {},
          readerLink,
        });
        saveStorage(state.storage);
        window.alert("Anyone with this link can read the messages in this profile.");
      } catch (error) {
        console.error("Failed to create profile access link.", error);
        els.readerLinkStatus.textContent = error?.message || "Failed to create the profile access link.";
        els.createReaderLinkButton.hidden = false;
      } finally {
        els.createReaderLinkButton.disabled = false;
      }
    }

    function configureCreateCard() {
      const noProfile = !state.storage.primaryInboxId;
      const isVisitorCreate = !state.route.isHomepageRoute && noProfile && Boolean(state.currentInbox);
      els.createCard.hidden = !isCreateCardActive();
      els.footerCard.hidden = !(state.route.isHomepageRoute && noProfile);
      els.createCardHeader.hidden = !isVisitorCreate;

        els.createCard.classList.add("create-card-visitor");
      } else {
        els.createCard.classList.remove("create-card-visitor");

      if (state.showCreateCard && !noProfile) {
        els.createInboxButton.textContent = "Create a new profile";
      } else if (isVisitorCreate) {
        els.createInboxButton.textContent = "Create profile link";
      } else {
        els.createInboxButton.textContent = state.route.isHomepageRoute ? "Create your profile link" : "Create your own profile";
      }

      syncCreateButtonState();
    }

    async function resolveCurrentInbox() {
      const route = state.route;

      if (route.invalidRoute) {
        throw new Error(route.hasAccessSegment ? "This profile access link is invalid." : "This profile link is invalid or unavailable.");
      }

      if (!route.inboxId) {
        return;
      }

      let currentEntry = state.storage.inboxes[route.inboxId] || null;
      let readerBundle = null;

      if (route.accessId) {
        readerBundle = await fetchReaderBundle(route.accessId);
        if (readerBundle.inboxId !== route.inboxId) {
          throw new Error("This profile access link does not match this profile.");
        }
      }

      let bundle;
      try {
        bundle = await fetchInboxBundle(route.inboxId, currentEntry?.uploadUrl || "");
      } catch (error) {
        if (route.isHomepageRoute && currentEntry) {
          console.warn("Homepage primary inbox could not be loaded. Clearing local primary inbox entry.", {
            inboxId: route.inboxId,
            uploadUrl: currentEntry?.uploadUrl || "",
            role: currentEntry?.role || "",
            error: error?.message || String(error),
          });
          delete state.storage.inboxes[route.inboxId];
          delete state.storage.blockedUsers[route.inboxId];
          if (state.storage.primaryInboxId === route.inboxId) {
            state.storage.primaryInboxId = "";
          }
          saveStorage(state.storage);
          state.currentInbox = null;
          state.currentPublicKey = "";
          state.currentPrivateKey = "";
          state.currentCanRead = false;
          state.currentOwnsInbox = false;
          state.currentProfile = null;
          state.currentBlogUsername = "";
          return;
        }
        throw error;
      }
      if (currentEntry?.publicKey && currentEntry.publicKey !== bundle.publicKey) {
        currentEntry = {
          ...currentEntry,
          publicKey: bundle.publicKey,
        };
      }

      if (readerBundle) {
        if (readerBundle.publicKey !== bundle.publicKey || !doesKeyPairMatch(bundle.publicKey, readerBundle.privateKey)) {
          throw new Error("This profile access link is invalid.");
        }

        upsertInbox({
          inboxId: readerBundle.inboxId,
          publicKey: bundle.publicKey,
          privateKey: readerBundle.privateKey,
          role: currentEntry?.role === "owner" ? "owner" : "reader",
          uploadUrl: currentEntry?.uploadUrl || "",
          createdAt: currentEntry?.createdAt || Date.now(),
          profile: currentEntry?.profile || {},
          readerLink: currentEntry?.readerLink || "",
          playMessageDing: currentEntry?.playMessageDing === true,
          playMessageDingVolume: currentEntry?.playMessageDingVolume,
        });
        saveStorage(state.storage);
        currentEntry = state.storage.inboxes[route.inboxId];
      }

      const verifiedPrivateKey = currentEntry?.privateKey && doesKeyPairMatch(bundle.publicKey, currentEntry.privateKey)
        ? currentEntry.privateKey
        : "";

      state.currentInbox = {
        inboxId: route.inboxId,
        uploadUrl: currentEntry?.uploadUrl || "",
        role: currentEntry?.role || "public",
      };
      state.currentPublicKey = bundle.publicKey;
      state.currentPrivateKey = verifiedPrivateKey;
      state.currentCanRead = Boolean(verifiedPrivateKey);
      state.currentOwnsInbox = Boolean(verifiedPrivateKey);
      state.currentProfile = bundle.profile;
      state.currentBlogUsername = sanitizeProfile(bundle.profile).commentsPluginUsername;

      if (state.currentBlogUsername) {
        const expectedBlogChannel = typeof window.secretInboxBuildBlogChannel === "function"
          ? window.secretInboxBuildBlogChannel(route.inboxId, state.currentBlogUsername)
          : "";
        if (window.secretInboxInitialBlog?.channel !== expectedBlogChannel) {
          try {
            sessionStorage.setItem(window.secretInboxBlogSeedKey, JSON.stringify({
              inboxId: route.inboxId,
              commentsPluginUsername: state.currentBlogUsername,
            }));
          } catch (error) {
            console.warn("Failed to store blog seed.", error);
          }
          state.initNeedsBlogReload = true;
          return;
        }
      } else {
        try {
          const seed = JSON.parse(sessionStorage.getItem(window.secretInboxBlogSeedKey) || "{}");
          if (normalizeId(seed?.inboxId) === route.inboxId) {
            sessionStorage.removeItem(window.secretInboxBlogSeedKey);
          }
        } catch (error) {
          sessionStorage.removeItem(window.secretInboxBlogSeedKey);
        }
      }

      if (currentEntry) {
        upsertInbox({
          ...currentEntry,
          publicKey: bundle.publicKey,
          profile: bundle.profile,
        });
        saveStorage(state.storage);
      }
    }

      els.visitorCard.hidden = true;
      els.messagesCard.hidden = true;
      els.blogCard.hidden = true;
      els.blogComposerCard.hidden = true;
      els.createCard.hidden = true;
      els.advancedPanel.hidden = true;
      els.footerCard.hidden = true;
      clearProfileTheme();

      try {
        await resolveCurrentInbox();
        if (state.initNeedsBlogReload) {
          window.location.reload();
          return;
        }
        await resolveProfileMediaState(state.currentProfile);
        state.initComplete = true;

        if (state.currentCanRead) {
          showShareCard();
          els.messagesCard.hidden = false;
        } else if (state.currentInbox) {
          els.visitorCard.hidden = false;

        renderBlog();
        renderVisitorCopy();
        configureCreateCard();
        renderProfile();

        if (state.currentCanRead && getCurrentInboxEntry()?.playMessageDing) {
          probeAudioPlayback().catch(() => {});

        renderInbox();

      } catch (error) {
        console.error("Failed to initialize profile.", error);
        state.initComplete = false;
        setLoading(false);
        showError(error?.message || "This profile could not be loaded.");
        renderBlog();
        renderVisitorCopy();
        configureCreateCard();
        renderProfile();
        renderAdvanced();
        updateAudioEnableButton();
      }
    }
