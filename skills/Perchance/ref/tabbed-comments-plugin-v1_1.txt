  function commentsPluginOnLoadHandler(comments) {

  function processCommentForUnseenCounts(comment) {
    // console.log("new comment:", comment);
    let channel = comment.channel;
    // initialize if we haven't yet:
    if(!knownMessageIdsByChannel[comment.channel]) knownMessageIdsByChannel[channel] = new Set();

    // increment the unseen count if we haven't seen this comment id before:
    if(!knownMessageIdsByChannel[channel].has(comment.id)) {
      knownMessageIdsByChannel[channel].add(comment.id);
      unseenCommentCountByChannel[channel] = (unseenCommentCountByChannel[channel] || 0) + 1;

      let tabIsVisible = false; // check if it's visible in any of the tabbedCommentsCtn elements
      for(let tabCtn of [...outerCtn.querySelectorAll(".tabbed-comments-plugin-ctn")]) {
        let tabContentEl = tabCtn.querySelector(`.tabbed-comments-plugin-content[data-channel='${channel}']`);
        if(tabContentEl && tabContentEl.offsetHeight !== 0) tabIsVisible = true;
      }
      if(tabIsVisible) { // set unseen count to 0 if tab is visible
        unseenCommentCountByChannel[channel] = 0;
      }
      updateUnseenCountHtml(channel);
    }
  }

  function isValidFolderName(folderName) {
    // examples of valid folder names:
    // foo
    // foo+channelname
    // foo+channelname+u:username1
    // foo+channelname+u:username1|username2

    let parts = folderName.split("+");
    if(parts.length > 3) return false;
    let name = parts[0] + (parts[1] ? `+${parts[1]}` : "");
    let rules = parts[2];
    
    let nameIsValid = name && /^[a-z0-9\-+]+$/.test(name) && name.split("+").length <= 2 && !name.endsWith("+");
    let rulesIfAnyAreValid = true;
    if(rules) rulesIfAnyAreValid = /^u:[a-z0-9|]+$/.test(rules) && !/\|\|/.test(rules) && !/:\|/.test(rules) && !rules.endsWith("|");
    return nameIsValid && rulesIfAnyAreValid;
  }
  
  function processCommentsForTempChannels(comments) {
    if(commentsChannels.allowCustomChannels === false) return;
    for(let comment of comments) {
      let mentionedChannelName = comment.message.match(/(\s|^)#[a-z0-9\-\+\|:]+(\s|$)/g)?.[0];
      if(mentionedChannelName && !isValidFolderName(window.generatorName+"+"+mentionedChannelName.trim().replace(/^#/, "").trim())) mentionedChannelName = "";
      if(mentionedChannelName && !/^[0-9]+$/.test(mentionedChannelName)) { // if it's all numbers, ignore
        mentionedChannelName = mentionedChannelName.trim().replace(/^#/, "").trim(); // remove the hash from the start and whitespace from start/end
        if(mentionedChannelName === "feedback") continue;
        if(!userIdToMentionedChannelsInLast2Minutes[comment.user.id]) userIdToMentionedChannelsInLast2Minutes[comment.user.id] = [];
        if(userIdToMentionedChannelsInLast2Minutes[comment.user.id].length >= 10) continue;
        // add it as a temporary channel which self-deletes after 2 minutes.
        // it upgrades to a real one when clicked.
        if(middlePanelObj) { // <-- just in case somehow hasn't initialized yet, or whatever

          // If they've got the tab *somewhere* (e.g. in side module), then we don't add it as a temporary channel.
          let alreadyExistsSomewhere = !!outerCtn.querySelector(`.tabbed-comments-plugin-ctn .tabbed-comments-plugin-header .tabbed-comments-plugin-tab[data-channel='${mentionedChannelName}']`);

          let { tab, alreadyExisted } = middlePanelObj.addCommentsChannel(mentionedChannelName);
          userIdToMentionedChannelsInLast2Minutes[comment.user.id].push(mentionedChannelName);
          
          if(!alreadyExisted && !alreadyExistsSomewhere) {
            temporaryMentionedChannels.add(mentionedChannelName);

            let newTabs = [];
            newTabs.push(tab);

            try { // because it's new code
              for(let obj of sidePanelObjs) {
                let { tab } = obj.addCommentsChannel(mentionedChannelName);
                newTabs.push(tab);
              }
            } catch(e) { console.error(e); }

            for(let newTab of newTabs) {
              newTab.dataset.tabbedCommentsPluginIsTemporaryMentionedChannel = "1";
              attachEventsToTemporaryMentionedChannelTab(newTab, mentionedChannelName);
            }
            // make its background blink for a couple of seconds:
            let blinkCount = 0;
            let blinkInterval = setInterval(() => {
              tab.style.background = blinkCount%2===0 ? "#043a9b" : "";
              blinkCount++;
              if(blinkCount > 5) clearInterval(blinkInterval);
            }, 300);
            setTimeout(() => {
              temporaryMentionedChannels.delete(mentionedChannelName);

              if(document.body.contains(tab) && tab.dataset.tabbedCommentsPluginIsTemporaryMentionedChannel) {
                for(let obj of [...sidePanelObjs, middlePanelObj]) {
                  obj.deleteCommentsChannel(mentionedChannelName);
                }
              }
            }, 1000*40);
          }
        }
      }
    }
  }

  function getAutoCommentNameFromNickname(nickname) {
    let nicknameWithoutNonNameCharacters = nickname.replace(/[\n\-:|()!{}*]/g, "").replace(/(?![*#0-9]+)[\p{Emoji}\p{Emoji_Modifier}\p{Emoji_Component}\p{Emoji_Modifier_Base}\p{Emoji_Presentation}]/gu, '');
    let words = nicknameWithoutNonNameCharacters.split(/[|/ \-})]/g).map(w => w.trim()).filter(w => w);
    while(words.slice(0, -1).join("_").length > 25) words.pop(); // we remove the last one for the length check in case it's a name like One Supercalifragilisticexpialidocious - better for userTag to be "OneSupercalif" than "One"
    if(words.slice(0, -1).join("_").length > 15) words.pop(); // we probably don't need the last word if all the earlier ones are over 15 characters
    let name = words.join("_").trim().toLowerCase();
    return name + (nickname.length > 25 ? "…" : "");
  }

        let name = getAutoCommentNameFromNickname(c.user.nickname);
        if(name) userTag = `${c.user.visualId.toUpperCase()}(${name})`
      }
      return `${userTag}: ${c.message.trim().replace(/\n{3,}/g, "\n\n")}`;

You are writing as a regular chat participant, not as an AI. When writing ${respondingUserTag}'s messages, you can ignore the off-topic/OOC/spam/commentary chat messages and focus on the roleplay messages that are actually a part of the story.     async function handleTabClick(event) {
      let clickedTab = event.target;
      if(!clickedTab.classList.contains('tabbed-comments-plugin-tab')) return; // they didn't click directly on a tab

      if(event.target.classList.contains('tabbed-comments-plugin-add-custom-channel')) {
        // They clicked on the "+" icon to add a new channel.
        // let channel = prompt("Choose a channel name. Note: You can type #channelname in the chat to share your channel.");
        let userInput = await getUserInput({
          channel: { type: "input:text", label: "Choose a channel name to add:", placeholder: "You can use letters, numbers and hyphens.", value: window.lastUsedAiSuggestPrompt || "" },
          tip:         { type: "html",     html: `<div style="opacity:0.8; font-size:75%;">
          <details>
            <summary style="cursor:pointer;">Tap here for tips</summary>
            <ul>
              <li>You can type <b>#channelname</b> in the chat to share your channel.</li>
              <li style="margin-top:0.5rem;">If you add <b>+u:alice|bob</b> at the end of a channel name, then only the users with usernames 'alice' and 'bob' will be able to comment on that channel. If your username is 'abc', and you want a personal 'blog' type channel that is publicly known but can't get spammed by others, then you could write <b>blog+u:abc</b> in the above text input, and only you will be able to comment on that channel.</li>
            </u>
            </details>
          </div>`, marginBottom:"0px" },
        }, { okText:"Add channel", cancelText:"Cancel", cancelOnBackgroundClick:true });
        let channel = userInput?.channel;
        if(channel) {
          let removeChannel = false;
          if(channel.startsWith("!")) {
            removeChannel = true;
            channel = channel.slice(1);
          }
          channel = channel.toLowerCase();
          channel = channel.replace(/[^a-z0-9\-:+|]/g, "");

          if(!channel) {
            alert("Channel names can only contain lower-case letters, numbers and hyphens.");
            return;
          }
          if(channel === "feedback") return;
          if(channel === "general") return;

          if(removeChannel) {
            deleteCommentsChannel(channel);
            let firstChannelTab = tabsHeaderEl.querySelector('.tabbed-comments-plugin-channel-tab');
            if(firstChannelTab) showTabByChannelName(firstChannelTab.dataset.channel);
          } else {
            addCommentsChannel(channel);
            showTabByChannelName(channel);
            alert(`The '${channel}' channel has been added. Type #${channel} in the chat to share it.`)
          }
        }
        return;
      } else if(event.target.classList.contains('tabbed-comments-plugin-channel-tab')) {
        showTabByChannelName(clickedTab.dataset.channel);
      }

    // Adds a new tab to the module
    function addTab(commentOptions) {
    
      commentOptions.width = "100%";
      commentOptions.height = "100%"; // only makes sense to set height via defaultChannelOptions, and that controls overall comments container height
      if(!commentOptions.customEmojis) commentOptions.customEmojis = defaultChannelOptions.customEmojis || defaultCustomEmojis;
      commentOptions.forceColorScheme = defaultChannelOptions.forceColorScheme ?? localStorage.forceColorScheme ?? null;
      
      let originalOnCommentHandler = commentOptions.onComment || defaultChannelOptions.onComment;
      let originalOnLoadHandler = commentOptions.onLoad || defaultChannelOptions.onLoad
      commentOptions.onComment = function(...args) {
        commentsPluginOnCommentHandler(...args); // for counting unseen, channel mentions, etc.
        try { if(originalOnCommentHandler) originalOnCommentHandler(...args); } catch(e) { console.error(e) } // userland code
      };
      commentOptions.onLoad = function(...args) {
        commentsPluginOnLoadHandler(...args); // for counting unseen, channel mentions, etc.
        try { if(originalOnLoadHandler) originalOnLoadHandler(...args);  } catch(e) { console.error(e) } // userland code
      };
    
      commentOptions.replacedDuringUpdate = true; // needed to allow multiple instances of the same channel (since we have multiple tabbed-comments-modules)
      for(let key of defaultChannelOptions.getAllKeys || Object.keys(defaultChannelOptions)) {
        if(!commentOptions[key] && !commentOptions.hasOwnProperty(key)) {
          // getter, because the default options could be dynamic (e.g. dark/light mode using localStorage), and this createTabbedCommentsPanel function is (as of writing) called every time the dark/light mode is switched.
          Object.defineProperty(commentOptions, key, {
            get: () => defaultChannelOptions[key],
          });
        }
      }
      
      let commentsPluginOutput = commentsPlugin(commentOptions);
      commentsChannelNameToObj[commentOptions.channel] = commentsPluginOutput;

      const tab = document.createElement('div');
      tab.className = 'tabbed-comments-plugin-tab tabbed-comments-plugin-channel-tab'; // note that non-channel tabs/buttons are possible - e.g. close tab button, close panel button
      tab.textContent = commentOptions.label || commentOptions.channel || "general";
      tab.innerHTML += `<span class="tabbed-comments-plugin-unseen-count">?</span>`;
      tab.dataset.channel = commentOptions.channel;

      const tabContent = document.createElement('div');
      tabContent.className = 'tabbed-comments-plugin-content';
      tabContent.innerHTML = commentsPluginOutput;
      tabContent.dataset.channel = commentOptions.channel;
      
      // for refreshing the comments if needed (e.g. when dark/light mode changes - see updateColorScheme function)
      tabContent.dataset.tabbedCommentsDirectParentOfCommentsPluginOutput = "1";
      tabContent.__commentOptionsUsed = commentOptions; // this should probably not be 'concrete' ones because we want a reference to the dynamic onces for if/when we need to reload it

      tabsHeaderEl.appendChild(tab);
      tabsContentEl.appendChild(tabContent);
      return {tab, tabContent};
    }

    onShowTab: ({channel}) => {
      middleCtnCurrentlyShownCommentsChannelName = channel;
      
      let channelData = commentsChannels[channel || "general"];
      if(!channelData) channelData = defaultChannelOptions;
      let allowBots = channelData.allowBots;
      if(channel === "" || channel === "general" || opts.allowBots === false) {
        if(allowBots === undefined) allowBots = false; // by default, don't allow bots on main/general/chat channel
      }
      autoCommentCtn.style.display = "none";
      if(allowBots !== false) {
        showAutoCommentAreaBtn.style.display = "";
      } else {
        showAutoCommentAreaBtn.style.display = "none";
      }
    },
  });
  middleCtnChannelNameToObj = middlePanelObj.commentsChannelNameToObj;
  middleCtn.appendChild(middlePanelObj.element);
  return outerCtn;

async getUserInput(fields, opts) =>
  fields = fields || {};
  opts = opts || {};
  const esc = s => String(s).replace(/[<>&]/g, m => ({"<":"&lt;","&":"&amp;",">":"&gt;"}[m]));
  const cancelText = esc(opts.cancelText || "Cancel");
  const okText = esc(opts.okText || "OK");

  return new Promise(resolve => {
    const overlay = Object.assign(document.createElement("div"), { tabIndex: 0 });
    overlay.style.cssText = `position:fixed;inset:0;z-index:99999999;display:grid;place-items:center;background-color:rgba(0,0,0,.65);font:16px/1.4 system-ui`;

    const modal = document.createElement("div");
    modal.style.cssText = `
      text-align:left !important;
      width:520px;max-width:97vw;
      border-radius:8px;padding:15px;
      background-color:light-dark(#fff,#222);
      color:light-dark(#000,#fff);
      box-shadow:0 2px 8px rgba(0,0,0,.2);
      display:flex;flex-direction:column;gap:12px;
      max-height:min(92vh, 900px);
    `;

    // Scrollable content
    const scrollWrap = document.createElement("div");
    scrollWrap.style.cssText = "overflow:auto;min-height:0;";
    const host = document.createElement("div");
    host.style.cssText = "display:block";
    scrollWrap.append(host);

    // Footer (Cancel left, OK right)
    const footer = document.createElement("div");
    footer.style.cssText = "display:flex;gap:8px;align-items:center;margin-top:4px;";
    const cancelBtn = document.createElement("button");
    cancelBtn.textContent = cancelText;
    cancelBtn.hidden = !!opts.hideCancel;
    cancelBtn.style.cssText = "padding:6px 16px;border:1px solid light-dark(#ccc,#555);border-radius:6px;background-color:light-dark(#f6f6f6,#333);color:inherit;cursor:pointer;";
    const okBtn = document.createElement("button");
    okBtn.textContent = okText;
    okBtn.autofocus = true;
    okBtn.style.cssText = "margin-left:auto;padding:6px 16px;border:none;border-radius:6px;background-color:light-dark(#1677ff,#2b87ff);color:#fff;cursor:pointer;";

    footer.append(cancelBtn, okBtn);
    modal.append(scrollWrap, footer);
    overlay.append(modal);

    const elMap = new Map();
    const inputCSS = "width:100%;max-width:100%;box-sizing:border-box;padding:8px;border:1px solid light-dark(#ccc,#555);border-radius:6px;background-color:light-dark(#fff,#333);color:inherit;";

    Object.entries(fields).forEach(([name, cfgRaw]) => {
      const cfg = cfgRaw || {};
      const t = (cfg.type || "input:text").toLowerCase();
      const wrap = document.createElement("div");
      wrap.style.cssText = `margin:0 0 ${cfg.marginBottom || "12px"}`;

      if (cfg.label) {
        const lab = document.createElement("div");
        lab.style.cssText = "margin:0 0 6px;display:flex;align-items:center;gap:6px;";
        const labText = document.createElement("div");
        labText.textContent = cfg.label || name;
        lab.append(labText);
        if (cfg.help) {
          const help = document.createElement("span");
          help.textContent = "i";
          help.title = cfg.help;
          help.style.cssText = "margin-left:auto;display:inline-grid;place-items:center;width:18px;height:18px;border-radius:50%;border:1px solid light-dark(#ccc,#555);font:12px/1 system-ui;opacity:.75;cursor:help;user-select:none";
          help.onclick = () => alert(cfg.help);
          lab.append(help);
        }
        wrap.append(lab);
      }

      let el = null, kind = "std";
      if (t === "textarea") {
        el = document.createElement("textarea");
        el.rows = cfg.rows || 4;
        el.style.cssText = inputCSS;
        if (cfg.placeholder != null) el.placeholder = cfg.placeholder;
        if (cfg.value != null) el.value = cfg.value;
      } else if (t === "select") {
        el = document.createElement("select");
        el.style.cssText = inputCSS;
        if (cfg.multiple) el.multiple = true;
        (cfg.options || []).forEach(opt => {
          const o = document.createElement("option");
          if (typeof opt === "object") { o.value = opt.value; o.textContent = opt.label ?? opt.value; }
          else { o.value = String(opt); o.textContent = String(opt); }
          el.append(o);
        });
        const v = cfg.value;
        if (el.multiple && Array.isArray(v)) Array.from(el.options).forEach(o => o.selected = v.includes(o.value));
        else if (v != null) el.value = v;
      } else if (t.startsWith("input:")) {
        el = document.createElement("input");
        el.type = t.split(":")[1] || "text";
        el.style.cssText = inputCSS;
        if (cfg.placeholder != null) el.placeholder = cfg.placeholder;
        if (cfg.min != null) el.min = cfg.min;
        if (cfg.max != null) el.max = cfg.max;
        if (cfg.step != null) el.step = cfg.step;
        if (el.type === "checkbox") el.checked = !!cfg.value;
        else if (cfg.value != null) el.value = cfg.value;
      } else if (t === "html") {
        const html = document.createElement("div");
        html.innerHTML = cfg.html || "";
        wrap.append(html);
        host.append(wrap);
        elMap.set(name, { cfg, el: html, wrap, kind: "html" });
        return;
      } else {
        const html = document.createElement("div");
        html.innerHTML = cfg.html || "";
        wrap.append(html);
        host.append(wrap);
        elMap.set(name, { cfg, el: html, wrap, kind: "html" });
        return;
      }

      el.name = name;
      wrap.append(el);
      host.append(wrap);
      elMap.set(name, { cfg, el, wrap, kind: "std" });
    });

    const finish = v => { overlay.remove(); resolve(v); };

    cancelBtn.onclick = () => finish(null);
    okBtn.onclick = () => {
      const out = {};
      elMap.forEach(({ el, kind, wrap }, name) => {
        if (kind === "html") {
          const dv = wrap.querySelector("[data-value]");
          out[name] = dv ? dv.getAttribute("data-value") : null;
        } else if (el.tagName === "SELECT") {
          out[name] = el.multiple ? Array.from(el.selectedOptions).map(o => o.value) : el.value;
        } else if (el.tagName === "TEXTAREA") {
          out[name] = el.value;
        } else if (el.tagName === "INPUT") {
          if (el.type === "checkbox") out[name] = el.checked;
          else if (el.type === "number" || el.type === "range") {
            const n = parseFloat(el.value); out[name] = Number.isFinite(n) ? n : null;
          } else out[name] = el.value;
        } else out[name] = null;
      });
      finish(out);
    };

    overlay.onkeydown = e => {
      if (e.key === "Escape") finish(null);
      else if (e.key === "Enter" && e.target.tagName !== "TEXTAREA") { e.preventDefault(); okBtn.click(); }
    };

    // Backdrop mousedown cancel only (not mouseup)
    if (opts.cancelOnBackgroundClick) {
      overlay.addEventListener("mousedown", e => {
        if (e.target === overlay) finish(null);
      });
    }

    document.body.append(overlay);
    overlay.focus({ preventScroll: true });

    const first = host.querySelector("input,textarea,select");
    if (first) setTimeout(() => first.focus({ preventScroll: true }), 0);
  });
